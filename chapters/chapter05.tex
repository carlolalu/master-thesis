The following chapter focuses on showing, in some sense, that the quantity

\begin{equation}
	\Norm{
		\drift^N(t, \mfsol_t^N)
		-
		\mfdrift^N(t, \mfsol_t^N)
	}_\infty
\end{equation}

is very unlikely to be big as $N$ (the number of amoebas) grows, where the attraction taken in consideration is slightly more general than the one considered till now (has an added Coulomb-type singularity).
More formally we would like to prove that, chosen some $\xi>0$ (with some extra condition)

\begin{equation}
	\PrbOf{
		\Norm{
			\drift^N(t, \mfsol_t^N)
			-
			\mfdrift^N(t, \mfsol_t^N)
	}_\infty
	\geq
	N^{-\xi}
	}
	\leq
	\const_n\
	(t^{4n} \: \vee \: 1)\
	N^{-\const_\xi n + 1}
\end{equation}

for all $n \in \N$, where $\const_n$ is a constant dependent on $n$ and independent from $\xi, N$, and $\const_\xi$ is a constant dependent only on $\xi$.
To prove it the idea is to notice that for the $k$-th particle

\begin{align*}
	N^{-\xi}\
	&\leq
	\Norm{
		\drift_k^N(t, \mfsol_t^N)
		-
		\mfdrift_k^N(t, \mfsol_t^N)
	}_\infty
	\\
	&=
	\Norm{
		\drift_k^N(t, \mfsol_t^N)
		-
		\ECond{\drift_k^N(t, \mfsol_t^N)}{ \mfamba_t}
	}_\infty
	\\
	&\leq
	\Norm{
		\drift_k^N(t, \mfsol_t^N)
		-
		\ECondTimes{\drift_k^N(t, \mfsol_t^N)}{ \mfamba_t, \:}
	}_\infty +
	\\
	&\qquad\quad
	+
	\Norm{
		\ECondTimes{\drift_k^N(t, \mfsol_t^N)}{ \mfamba_t, \:}
		-
		\ECond{\drift_k^N(t, \mfsol_t^N)}{ \mfamba_t}
	}_\infty
\end{align*}

and therefore, using this knowledge together with \ref{app:pA-pB-less-than}, we can obtain

\begin{equation}\label{ch5:sketch-dividi-et-impera}
	\begin{aligned}
		&
		\PrbOf{
			\Norm{
				\drift^N(t, \mfsol_t^N)
				-
				\mfdrift^N(t, \mfsol_t^N)
			}_\infty
			\geq
			N^{-\xi}
		}
		= 
		\PrbOf{
			\sup_{1\leq k \leq N} 
			\Norm{
				\drift_k^N(t, \mfsol_t^N)
				-
				\mfdrift_k^N(t, \mfsol_t^N)
			}_\infty
			\geq
			N^{-\xi}\
		}
		\leq
		\\
		&\leq
		\Prb \Bigg(
			\sup_{1\leq k \leq N} 
			\Norm{
				\drift_k^N(t, \mfsol_t^N)
				-
				\ECondTimes{\drift_k^N(t, \mfsol_t^N)}{ \mfamba_t, \:}
			}_\infty
			+
			\\
			&\qquad\quad
			+
			\sup_{1\leq k \leq N} 
			\Norm{
				\ECondTimes{\drift_k^N(t, \mfsol_t^N)}{ \mfamba_t, \:}
				-
				\ECond{\drift_k^N(t, \mfsol_t^N)}{ \mfamba_t}
			}_\infty
			\geq
			N^{-\xi}\
			+N^{-\xi}\
		\Bigg)
		\\
		&\leq
		\overbrace{
			\PrbOf{
				\sup_{1\leq k \leq N} 
				\Norm{
					\drift_k^N(t, \mfsol_t^N)
					-
					\ECondTimes{\drift_k^N(t, \mfsol_t^N)}{ \mfamba_t, \:}
				}_\infty
				\geq
				N^{-\xi}
		}}^{\text{Term 1}}\ +
		\\
		&\qquad\quad
		+
		\underbrace{
			\PrbOf{
				\sup_{1\leq k \leq N} 
				\Norm{
					\ECondTimes{\drift_k^N(t, \mfsol_t^N)}{ \mfamba_t, \:}
					-
					\ECond{\drift_k^N(t, \mfsol_t^N)}{ \mfamba_t}
				}_\infty
				\geq
				N^{-\xi}
			}
		}_{\text{Term 2}}
	\end{aligned}
\end{equation}

and consequently we can bound separately the two addends.
This was the "dividi" part of the usual strategy "dividi et impera".
The first section is an explains the generalization of the attraction $\attraction$ and shows some estimate about it, the two central sections treat respectively the first and the second term, and the final fourth section composes all the pieces together in a ready-to-use tool.

\section{Necessary premises}

Only for this chapter we will use a more general case for the attraction: we will use an additional term which also admits a cut-off Coulomb-type of singularity.

\begin{definition}[Coulomb-type attraction-potential, generalized attraction and drift]\label{ch5:def:generalized-attr&drift}
	Let $\alpha \geq 0$.
	Let $\cutoffSinAttraction = \cutoffSinAttraction (N, \alpha) : \R^d \longrightarrow \R^d$ be a function such that
	\begin{equation}
		0
		\leq 
		\norm{\cutoffSinAttraction}
		\leq 
		\const_\curlywedge \left( 
		\frac{1}{\norm{x}} \wedge N^\alpha
		\right)
		=
		\left\{
		\begin{aligned}
			&\frac{\const_\curlywedge}{\norm{x}}			&& x \in \{ \norm{x} \geq N^{-\alpha}\}
			\\
			&\const_\curlywedge\:N^{\alpha}						&& \text{otherwise}
		\end{aligned}
		\right.
	\end{equation}
	for some positive constant $\const_\curlywedge$,	\ie such that it is less than a cut-off Coulomb-type singularity, with the cut-off of parameter $N^{\alpha}$.
	Then we define the generalized attraction with cut-off of parameters $\alpha$ as $\genAttraction := \genAttraction(N, \alpha, \const_\curlywedge) := \attraction + \cutoffSinAttraction(N, \alpha)$, and the generalized drift as the drift of \ref{ch2:def:drift-&-diffusion} with $\attraction$ substituted with $\genAttraction$, \ie
	\begin{equation}
		\left\{
		\begin{aligned}
			{}^{\genAttraction}\drift_k(t,x, (\hmactive^k_t)_k) &:= \drift_p(t,x) := \frac{1}{N} \sum _{p = 1}^{N} \sum_{g = 1}^{\hmactive_t^p} \genAttraction(x^k - x^{p,g})
			&& 1\leq k \leq N
			\\
			{}^{\genAttraction}\drift_{p,g}(t,x) &:= 0
			&& \text{if } \exists (p,g)
		\end{aligned}
		\right.
	\end{equation}
	Analogous is the case for the generalized mean-field drift:
		\begin{equation}
		\left\{
		\begin{aligned}
			{}^{\genAttraction}\mfdrift_k(t,x, (\hmactive^k_t)_k) &:= \mfdrift_p(t,x) := 
			\lambda
			\int
				(\genAttraction \star \cdens_t^\tau) (x)
			d{\:\tau}
			&& 1\leq k \leq N
			\\
			{}^{\genAttraction}\mfdrift_{p,g}(t,x) &:= 0
			&& \text{if } \exists (p,g)
		\end{aligned}
		\right.
	\end{equation}
\end{definition}

For the sake of shortness we will simply write $\drift,\mfdrift$ for ${}^{\genAttraction}\drift, {}^{\genAttraction}\mfdrift$ all along this subsection, where such generalized attraction is used.

Before proceeding to the law of large numbers we need to gather some preliminary lemmas.
The first is a simple result of \autocite{peter-boers}, that we simply report here.

\begin{lemma}\label{ch5:lemma:<-1+infty-norm}
	For a function $f$ define the eventually infinite quantities
	\begin{gather*}
		\Norm{f}_{1 \wedge \infty} := 
		\inf_{f_1 + f_\infty = f} 
		\left( 
		\Norm{f_1}_1 + \Norm{f_\infty}_\infty
		\right),
		\qquad
		\qquad
		\Norm{f}_{1 \vee \infty} := 
		\Norm{f}_1 + \Norm{f}_\infty
	\end{gather*}
	Then for arbitrary functions $f,g$ it is true that
	\begin{equation}
		\Norm{f \star g}_\infty
		\leq
		\Norm{f}_{1 \wedge \infty} \cdot \Norm{g}_{1 \vee \infty}
	\end{equation}
\end{lemma}

\begin{proof}
	Let $g_1 + g_\infty = g$ be a decomposition of $g$.
	Then
	\begin{align*}
		&\Norm{f \star g}_\infty 
		\leq 
		\Norm{f \star g_1}_\infty + \Norm{f \star g_1}_1
		\leq
		\\
		&\leq
		\Norm{f}_\infty \Norm{g_1}_1 + \Norm{f}_1 \Norm{g_\infty}_\infty
		\\
		&\leq
		(\Norm{f}_\infty + \Norm{f}_1) ( \Norm{g_1}_1 + \Norm{g_\infty}_\infty)
		= \Norm{f}_{1 \vee \infty}	( \Norm{g_1}_1 + \Norm{g_\infty}_\infty)
	\end{align*}
	Since this is valid for every decomposition $g_1 + g_\infty = g$, we have our result.
\end{proof}

The second lemma that we need is a simple evaluation of these quantities for the generalized attraction.

\begin{lemma}\label{ch5:lemma:eval-norm-gen-attr}
	Let $\genAttraction$ be the generalized attraction, then we have that there exists a constant $\const_{\attraction,\curlywedge}$ depending only on $\attraction, \const_\curlywedge$ such that
	\begin{align}
		\Norm{\genAttraction}_{1 \wedge \infty}, \:
		\Norm{\genAttraction^2}_{1 \wedge \infty} 
		\leq \const_{\attraction,\curlywedge}
	\end{align}
\end{lemma}

\begin{proof}
	Let us first operate for the power $1$.
	Choose the decomposition 
	\begin{equation}
		\genAttraction = \attraction + \cutoffSinAttraction = 
		\underbrace{ 
			\attraction + \cutoffSinAttraction \1_{\norm{x} \leq 1}
		}_{=:(\genAttraction)_1}
		+
		\underbrace{
			\cutoffSinAttraction \1_{\norm{x} > 1}
		}_{=:(\genAttraction)_\infty}
	\end{equation}
	
	[[put here a tikzfigure with everythign]]
	
	Then we have that
	
	\begin{align*}
		&\Norm{(\genAttraction)_1}_1
		\leq
		\Norm{\attraction}_1 + \Norm{\cutoffSinAttraction \1_{\norm{x} \leq 1}}_1
		=
		\\
		&=
		\Norm{\attraction}_1 + \Norm{\const_\curlywedge N^\alpha \1_{0 \leq \norm{x} \leq N^{-\alpha}}}_1 + \Norm{\frac{\const_\curlywedge}{\norm{x}} \1_{N^{-\alpha} \leq \norm{x} \leq 1}}_1
		\\
		&\footnotemark[1]=
		\Norm{\attraction}_1 + \const_\curlywedge N^{\alpha} \cdot \lebsge (\Ball^{d-1}[N^{-\alpha}]) + \int_{N^{-\alpha}}^1 \: \frac{\const_\curlywedge}{\norm{x}} \: dx
		\\
		& \leq
		\Norm{\attraction}_1 
		+ \const_\curlywedge N^{\alpha} 
		\max \left\{
		\pi (N^{-\alpha})^2
		,\frac{4}{3} \pi (N^{-\alpha})^3
		\right\}
		+
		\\
		&\qquad
		+
		\const_\curlywedge
		\max \left\{
		\int_{N^{-\alpha}}^1  \: \frac{1}{r} r\: dr \:2\pi
		,\int_{N^{-\alpha}}^1  \: \frac{1}{r} r^2\: dr \: 2 \pi \int_0^\pi \sin(\theta) d\theta
		\right\}
		\\
		& \leq 
		\Norm{\attraction}_1 
		+ \const_\curlywedge N^{\alpha-2\alpha} 2 \pi
		+ \const_\curlywedge 2 \pi \cdot 2 \max \left\{
		\int_0^1 dr
		,\int_0^1  r\: dr
		\right\}
		\\
		& \leq
		\Norm{\attraction}_1
		+ \const_\curlywedge 6\pi 
	\end{align*}
	\footnotetext[1]{$\lebsge(\Ball^{d-1}[N^{-\alpha}])$ indicates the Lebesgue measure of the $d-1$-ball of radius $N^{-\alpha}$}
	
	And obviously we have that
	
	\begin{equation*}
		\Norm{\cutoffSinAttraction \1_{\norm{x} > 1}}_\infty 
		\leq 
		\Norm{\frac{\const_\curlywedge}{\norm{x}} \1_{\norm{x} > 1}}_\infty  = \const_\curlywedge
	\end{equation*}
	
	Then the first inequality is proven.
	
	Let us proceed with the power $2$.
	Choose this decomposition:
	
	\begin{align*}
		\genAttraction^2 
		&= 
		\left(
		\attraction + \cutoffSinAttraction
		\right)^2
		= \attraction^2 + 2 \attraction \cutoffSinAttraction + \cutoffSinAttraction^2 
		\\
		& = 
		\underbrace{ 
			\attraction^2 + 2 \attraction \cutoffSinAttraction \1_{\norm{x} \leq 1}
		}_{=:(\genAttraction^2)_1}
		+
		\underbrace{
			\attraction \cutoffSinAttraction \1_{\norm{x} > 1}
			+ \cutoffSinAttraction^2 \1_{\norm{x} > 1}
		}_{=:(\genAttraction^2)_\infty}
	\end{align*}
	
	Then, proceeding as before, we have that
	
	\begin{equation*}
		\Norm{(\genAttraction^2)_1}_1 
		\leq
		\Norm{\attraction^2}_1 + 2 \Norm{\attraction}_\infty \int \norm{\cutoffSinAttraction}  \1_{\norm{x} \leq 1}
		\leq \Norm{\attraction^2}_1 + 2 \Norm{\attraction}_\infty \const_\curlywedge 6\pi
	\end{equation*}
	
	And regarding the other term:
	
	\begin{equation*}
		\Norm{\attraction \cutoffSinAttraction \1_{\norm{x} > 1}
			+ \cutoffSinAttraction^2 \1_{\norm{x} > 1}}_\infty 
		\leq 
		\Norm{\frac{\const_\curlywedge \attraction}{\norm{x}} \1_{\norm{x} > 1}}_\infty + \Norm{\frac{\const_\curlywedge^2}{\norm{x}^2} \1_{\norm{x} > 1}}_\infty  = \const_\curlywedge\Norm{\attraction}_\infty + \const_\curlywedge^2
	\end{equation*}
\end{proof}


\section{The first part of the law of large numbers}

Now we can finally approach the first part of the law of large numbers.
The proof is classical law of large numbers, following for example \autocite{ana}.
It uses a notation akin to the relabelled model defined in \ref{ch2:model-reformulation}.

\begin{proposition}[The first part of the law of large numbers]\label{ch5:prop:LLN1}
	
	Let $\hmactive_t$ be the spawning Poisson process of \ref{ch2:model-reformulation}. Let $\mfsol^N$ be a \rv that under $\PrbCondTimes{\cdot}{}$ takes the form
	\begin{equation}
		\mfsol^N = (\mfamba^1, \ldots, \mfamba^N, \mfcamp^1, \ldots, \mfcamp^{\hmactive_t}) \in (\R^d)^{\NumParticles_t} =  (\R^d)^{N + \hmactive_t} 
	\end{equation}
	Let $\drift^{N,k}(t, \cdot)$ be the \emph{generalized} drift of parameter $\alpha$ defined in \ref{ch5:def:generalized-attr&drift}.
	Assume that:
	\begin{enumerate}
		\item All the coordinates of $\mfsol^N$ are $\indep$ under the p. measure $\PrbCondTimes{}{}$.
		\item All the coordinates $\mfamba^p$ of $\mfsol^N$ are \iid under the p. measure $\PrbCondTimes{}{}$.
		\item The densities of all coordinates of $\mfsol^N$ are uniformly bounded in $\Lpspace^{\infty}$, \ie
		\begin{equation*}
			\exists \const_\dens>0.\
			\forall p,m.\
			\Norm{\cdens^m}_\infty, \Norm{\adens^p}_\infty < \const_\dens
		\end{equation*}
		where $\adens^p$ and $\cdens^m$ represent the laws of $\mfamba^p$ and $\mfcamp^m$ under $\PrbCondTimes{}{}$ respectively.
	\end{enumerate}
	Consider $\xi >0$ be such that $0<\xi + \alpha < \frac{1}{2}$ and define the sets
	\begin{equation*}
		\begin{aligned}
			S^{N,\xi}_\amba
			&:= \left\{ \sup_{1\leq k \leq N} 
			\norm{
				\drift_k^N(t, \mfsol^N) 
				- \ECondTimes{
					\drift^N_k(t,\mfsol^N)
				}{\mfamba^k,}
			}_\infty \geq N^{-\xi} 
			\right\}
			\\
			S^{N, \xi}
			&:= \left\{ \sup_{1\leq k \leq N} 
			\norm{
				\drift_k^N(t, \mfsol^N) 
				- \ECondTimes{
					\drift^N_k(t,\mfsol^N)
				}{}
			}_\infty \geq N^{-\xi} 
			\right\}
		\end{aligned}
	\end{equation*}
	
	Then, signing as $S_\ast^{N, \xi}$ either of the two sets, for $S_\ast^{N, \xi}$ is valid the laws of large numbers, \ie $\exists \overline{N} = \overline{N}(\alpha, \curlywedge)$ such that for all $\const_\xi$ such that $0 < \const_\xi \leq 1 - 2(\alpha + \xi)$ and for all $n\in\N$ exists $\const_n = \const_{\attraction, \curlywedge, \lambda, n}  >0$ and an increasing positive continuous function $\Fup_n$ (independent from $N, \alpha,\xi$)  so that it is valid:
	\begin{align}
		& \forall N> \overline{N}.	&	&\PrbOf{S^{N,\xi}_\ast}{}
		\leq 
			\const_n\
			\Fup_n(\const_\dens)\
			\Fup_n(t)\
			N^{-\const_\xi n + 1}
		&& \label{ch5:LLN1:bound}
	\end{align}
\end{proposition}


\begin{remark}\label{ch5:assumption-t,lambda>1-proofs}
	All along the subsequent proof it will be assumed that $t, \lambda \geq 1$.
	This is because in general for $0 < p <q$ we have that
	
	\begin{equation*}
		t^p \leq (t \vee 1)^q
	\end{equation*}
	
	and this clutters the notation and creates only non-necessary headaches for a function of $t$ which cannot explode (all the powers involved are positive integers).
\end{remark}

We will now prove such result.
For the sake of shortness, together with the convention at \ref{app:notation}, we will use the subsequent notation:
	
	\begin{equation*}
		S^{N, \xi}_\ast := \: \text{either of $S^{N, \xi}_\amba$ and $S^{N, \xi}$}
	\end{equation*}
	
	Notice that 
	
	\begin{align*}
		\PrbOf{S^{N,\xi}_\ast}
		& =\PrbOf{
			\sup_{1\leq k \leq N} 
			\norm{
				\drift_k^N(t, \mfsol^N) 
				- \ETFoot{\ast}{
					\drift^N_k(t,\mfsol^N)
				}
			}_\infty \geq N^{-\xi} 
		}
		\\
		& \leq \sum_{k=1}^{N} \sum_{i=1}^{d}
		\PrbOf{\norm{
				\drift^N_{k;i}(t, \mfsol^N) 
				- \ETFoot{\ast}{
					\drift^N_{k;i}(t,\mfsol^N)
				}
			} \geq N^{-\xi} 
		}
		\\
		&\leq
		N \cdot 3 \cdot
		\max_{k,i}
		\:
		\underbrace{
			\PrbOf{\norm{
					\drift^N_{k;i}(t, \mfsol^N) 
					- \ETFoot{\ast}{
						\drift^N_{k;i}(t,\mfsol^N)
					}
				} \geq N^{-\xi} 
			}
		}_{\Ch(k;i)}
	\end{align*}
	where recall that $\drift_{k;i}$ indicates the $i$-th coordinate for the $d$-dimensional drift of the $k$-th particle.
	Since the expression has the same form for each coordinate $(k;i)$, we assume that the maximum of $\Ch(k;i)$ is reached in $(1;1)$, and tame with a bound independent from such assumption.
	
	We shall begin by applying Markov inequality with function $f(x):= \norm{x}^{2n}$ for a generic $n \in \N$.
	\begin{align*}
		&=
		3N \cdot
		\PrbOf{\norm{
				\drift^N_{1;1}(t, \mfsol^N) 
				- \ETFoot{\ast}{
					\drift^N_{1;1}(t,\mfsol^N)
				}
			} \geq N^{-\xi} 
		}
		\\
		&\leq \frac{3N}{ (N^{-\xi})^{2n} } \E{
			\norm{
				\drift^N_{1;1}(t, \mfsol^N) 
				- \ETFoot{\ast}{
					\drift^N_{1;1}(t,\mfsol^N)
				}
			}^{2n}
		}
		\\
		& \footnotemark[1]=
		3
		N^{\xi 2n + 1} \E{
			\norm{
				\frac{1}{N} 
				\sum _{m = 1}^{\hmactive_t} 
				(\genAttraction)_1(\mfamba^1 - \mfcamp^m)
				- \frac{1}{N} \sum _{m = 1}^{\hmactive_t} 
				\ETFoot{\ast}{
					(\genAttraction)_1(\mfamba^1 - \mfcamp^m)
				}
			}^{2n}
		}
		\\
		& = 3 N^{\xi 2 n + 1} \frac{1}{N^{2n}} 
		\E{
			\norm{
				\sum _{m = 1}^{\hmactive_t} 
				\left[
				(\genAttraction)_1(\mfamba^1 - \mfcamp^m)
				- \ETFoot{\ast}{ \left( (\genAttraction)_1 \star \cdens^m \right) (\mfamba^1)}
				\right]
			}^{2n}
		}
		\\
		&= 3 N^{(\xi -1) 2 n + 1}
		\E{
			\norm{
				\sum _{m = 1}^{\hmactive_t} 
				\left[
				\underbrace{
					(\genAttraction)_1(\mfamba^1 - \mfcamp^m)
				}_{:=\Psi_m}
				-
				\underbrace{
					\ETFoot{\ast}{ \left( (\genAttraction)_1 \star \cdens^m \right) (\mfamba^1)}
				}_{=\ETFoot{\ast}{\Psi_m}=: \psi_m^\ast}
				\right]
			}^{2n}
		}
		=
	\end{align*}
	\footnotetext[1]{{Here $(\genAttraction)_1$ indicates the first coordinate of $\genAttraction$, and not a first part of any decomposition of $\genAttraction$.}}
	
	Setting $\Zhi_m := \norm{\Psi_m - \psi_m^\ast}$, we can apply the multinomial formula:
	
	\begin{align}
		\left( \sum_{m=1}^M \Zhi_m \right)^{2n}
		&= \sum_{\beta_1 + \ldots + \beta_M =2n} 
		\:
		\frac{(2n)!}{\beta_1! \ldots \beta_M!} \prod_{m=1}^M \Zhi_m^{\beta_m}
		\\
		&= \sum_{\beta_1 + \cdots + \beta_M =2n} 
		\:
		\frac{(2n)!}{(\beta)!} \prod_{m=1}^M \Zhi_m^{\beta_m}
	\end{align}
	
	where $\beta$ is the multi-index $(\beta_1, \ldots, \beta_M)$ and $(\beta)! := \beta_1!\ldots \beta_M!$.
	Then we obtain
	
	\begin{align*}
		&= 3 N^{(\xi -1) 2 n + 1}
		\E{
			\sum_{\beta_1 + \cdots + \beta_{\hmactive_t} =2n} \
			\frac{(2n)!}{(\beta)!}
			\prod_{m=1}^{\hmactive_t}
			\norm{
				\Psi_m - \psi_m^\ast
			}^{\beta_m}
		}
		\\
		& = 3 N^{(\xi -1) 2 n + 1}
		\E{
			\ETFoot{\ast}{
				\sum_{\beta_1 + \cdots + \beta_{\hmactive_t} =2n} \
				\frac{(2n)!}{(\beta)!}
				\prod_{m=1}^{\hmactive_t}
				\norm{
					\Psi_m - \psi_m^\ast
				}^{\beta_m}
			}
		}
		\\
		& = 3 N^{(\xi -1) 2 n + 1}
		\E{
			\sum_{\beta_1 + \cdots + \beta_{\hmactive_t} =2n}
			\frac{(2n)!}{(\beta)!}
			\ETFoot{\ast}{
				\prod_{m=1}^{\hmactive_t}
				\norm{
					\Psi_m - \psi_m^\ast
				}^{\beta_m}
		}}
		\\
		& = 
		3(2n)!\
		N^{(\xi -1) 2 n + 1}

	\begin{claim}\label{ch5:proof-of-LLN1:claim:bound-on-Psi-psi}
		There exists a constant $\const_{\attraction,\text{dens}, \curlywedge}>0$ depending on $\attraction,\const_{\text{dens}}$,  and there exists $\overline{N}$ depending on $\const_\curlywedge,\attraction, \alpha$ such that
		\begin{equation}
			\forall N > \overline{N}.\qquad
			\ETFoot{\ast}{
				\norm{
					\Psi_m - \psi_m^\ast 
				}^{\beta_m}
			}
			\leq 
			\left\{
			\begin{aligned}
				& 1		&&\text{if} \quad \beta_m = 0
				\\
				& 0 		&&\text{if} \quad \beta_m = 1
				\\
				&(\const_{\attraction, \curlywedge} 
				\Fup(\const_\dens)
				N^\alpha)^{\beta_m -2}	\quad		&&\text{otherwise}
			\end{aligned}
			\right.
		\end{equation}
	\end{claim}
	
	\begin{proof}[Proof of the claim]
		If $\beta_m =0$ we have
		
		\begin{equation*}
			\ETFoot{\ast}{
				\norm{
					\Psi_m
					- \psi_m^\ast
				}^0
			} = \ETFoot{\ast}{ \ETFoot{1}{ 1} } = 1
		\end{equation*}
		
		If instead $\beta_m =1$ we have:
		\begin{equation*}
			\ETFoot{\ast}{\norm{\Psi_m - \psi_m^\ast }^1 }
			=
			\ETFoot{\ast}{
				\ETFoot{1}{
					\norm{
						\Psi_m
						- \psi_m^\ast
					}
			}}
			= 	\ETFoot{\ast}{
				\norm{
					\Psi_m
					- \ETFoot{\ast}{\Psi_m}
				}
			}	= 0
		\end{equation*}
		
		We need then to take care only of the case $\beta_m \geq 2$.
		Here we have
		
		\begin{align*}
			\ETFoot{\ast}{\norm{\Psi_m - \psi_m^\ast }^{\beta_m} }
			& =
			\ETFoot{\ast}{\norm{\Psi_m - \psi_m^\ast }^{\beta_m - 2} \cdot \norm{\Psi_m - \psi_m^\ast }^{2}}
			\\
			&\leq \Norm{\Psi_m - \psi_m^\ast }^{\beta_m - 2}_\infty \cdot \ETFoot{\ast}{\norm{\Psi_m - \psi_m^\ast }^{2} }
		\end{align*}
		
		Then we can dominate the two terms separately.
		
		\begin{align*}
			\Norm{\Psi_m - \psi_m^\ast }_\infty 
			&= \Norm{(\genAttraction)_1(\mfamba^1 - \mfcamp^m)
				-
				\ETFoot{\ast}{ \left( (\genAttraction)_1 \star \cdens^m \right) (\mfamba^1)}
			}_\infty
			\\
			&=\Norm{\genAttraction}_\infty + \Norm{\genAttraction \star \cdens^m}_\infty
			\\
			& \leq \Norm{\genAttraction}_\infty + \Norm{\genAttraction}_\infty \Norm{\cdens^m}_1
			\\
			& = 2 \Norm{\genAttraction}_\infty = 2 (\Norm{\attraction}_\infty +  \Norm{\cutoffSinAttraction}_\infty )
			\\
			& \leq 2 (2 \Norm{\cutoffSinAttraction}_\infty ) =  \const_\curlywedge 4 N^{\alpha}
		\end{align*}
		
		where the last inequality is valid if $N$ is big enough, \ie $N > \overline{N}(\alpha, \curlywedge, \attraction)$.
		Then
		
		\begin{align*}
			\Norm{\Psi_m - \psi_m^\ast }_\infty^{\beta_m - 2} 
			\leq 
				(\const_\curlywedge 4 )^{\beta_m -2} N^{\alpha (\beta_m -2)}
			=
				(\const_\curlywedge 4 N^\alpha)^{\beta_m -2}
		\end{align*}
		
		Let us now check the other term:
		
		\begin{align*}
			&\ETFoot{\ast}{\norm{\Psi_m - \psi_m^\ast }^{2} } =
			\\
			&= \ETFoot{\ast}{
				\ETFoot{1}{
					\norm{
						(\genAttraction)_1(\mfamba^1 - \mfcamp^m)
						- \psi_m^\ast }^{2}
				}
			}
			\\
			&=
			\ETFoot{\ast}{
				\int
				\left(
				(\genAttraction)_1(\mfamba^1 - x)
				- \psi_m^\ast \right)^{2}
				\:
				\cdens^m(x)
				\:
				dx
			}
			\\
			& = \ETFoot{\ast}{
				\int
				(\genAttraction)_1^2(\mfamba^1 - x)
				\cdens^m(x)
				dx
				- \int 2
				(\genAttraction)_1(\mfamba^1 - x)
				\psi_m^\ast
				\cdens^m(x)
				dx
				+
				(\psi_m^\ast)^2
				\int
				\cdens^m(x)
				dx
			}
			\\
			&= \ETFoot{\ast}{
				\left( (\genAttraction)_1^2 \star \cdens^m \right) (\mfamba^1)
				- 2 \psi_m^\ast \left( (\genAttraction)_1 \star \cdens^m \right) (\mfamba^1)
				+(\psi_m^\ast)^2
			}
			\\
			& \leq \ETFoot{\ast}{
				\Norm{(\genAttraction)_1^2 \star \cdens^m}_\infty
				+ 2 \psi_m^\ast \Norm{(\genAttraction)_1 \star \cdens^m}_\infty
				+(\psi_m^\ast)^2
			}
			\\
			&= \Norm{(\genAttraction)_1^2 \star \cdens^m}_\infty
			+ 2 \psi_m^\ast \Norm{(\genAttraction)_1 \star \cdens^m}_\infty
			+(\psi_m^\ast)^2 
			\\
			&\leq 
			\Norm{(\genAttraction)_1^2 \star \cdens^m}_\infty
			+ 2\Norm{(\genAttraction)_1 \star \cdens^m}_\infty^2
			+ \Norm{(\genAttraction)_1 \star \cdens^m}_\infty^2
			\\
			& = 
			\Norm{(\genAttraction)_1^2 \star \cdens^m}_\infty
			+ 3\Norm{(\genAttraction)_1 \star \cdens^m}_\infty^2
			\qquad \text{}
			\\
			&\footnotemark[1]
			\leq
			\Norm{\genAttraction^2}_{1 \wedge \infty}
			\Norm{\cdens^m}_{1 \vee \infty}
			+ 3 
			\left(
			\Norm{\genAttraction}_{1 \wedge \infty}
			\Norm{\cdens^m}_{1 \vee \infty}
			\right)^2
			\\
			& =
			\const_{\curlywedge,\attraction} (1 + \const_{\text{dens}})
			+ 3 \const_{\curlywedge,\attraction}^2 (1 + \const_{\text{dens}})^2
			\\
			&
			\leq
				(\const_{\curlywedge,\attraction} + 3 \const_{\curlywedge,\attraction}^2)\
				[1 + \const_{\text{dens}} + (1 + \const_{\text{dens}})^2 ]
			\\
			& = 
				\const_{\attraction, \curlywedge}\
				\Fup(\const_\dens)
			\leq 
				[
					\const_{\attraction, \curlywedge}\Fup(\const_\dens)
				]^{\beta_m -2}
		\end{align*}
		\footnotetext[1]{By \ref{ch5:lemma:<-1+infty-norm} and \ref{ch5:lemma:eval-norm-gen-attr}}
		
		Then we have
		
		\begin{align*}
			\ETFoot{\ast}{\norm{\Psi_m - \psi_m^\ast }^{\beta_m} }
			&\leq \Norm{\Psi_m - \psi_m^\ast }^{\beta_m - 2}_\infty \cdot \ETFoot{\ast}{\norm{\Psi_m - \psi_m^\ast }^{2} }
			\\
			&\leq
				(\const_\curlywedge 4 N^\alpha)^{\beta_m -2}
				[
					\const_{\attraction, \curlywedge}\Fup(\const_\dens)
				]^{\beta_m -2}
			&=
					[
						\const_{\attraction, \curlywedge} 
						\Fup(\const_\dens)
						N^\alpha
					]^{\beta_m -2}
		\end{align*}
		And this concludes the proof of the claim.
	\end{proof}

		\\
		&\leq
		3(2n)!\
		N^{(\xi -1) 2 n + 1}
		\E{
			\sum_{\beta_1 + \cdots + \beta_{\hmactive_t} =2n}\
			\prod_{m=1}^{\hmactive_t}

				\norm{
					\Psi_m - \psi_m^\ast

			}

		\\
		&\leq
		3(2n)!\
		N^{(\xi -1) 2 n + 1}
		\E{
			\sum_{\beta_1 + \cdots + \beta_{\hmactive_t} =2n}\

	We can assume that $\Iu \geq 1$ (eventually modulate the value $\const_{\attraction,\text{dens}, \curlywedge} $ accordingly).
	
	\begin{claim}\label{ch5:proof-of-LLN1:sum-prod-Iu-inequality}
		$\Iu \geq 1$ is independent from the multi-index $\beta$, therefore
		\begin{equation}
			\sum_{\beta_1 + \cdots + \beta_M =2n}\
			\prod_{m=1}^{M}\
			\left(
			\1_{\beta_m = 0} + \1_{\beta_m \geq 2}
			\Iu^{\beta_m -2} 
			\right)
			\leq
			(2n)^n\
			(M)^n\
			\Iu^{2n-2}
		\end{equation}
	\end{claim}
	
	\begin{proof}[Proof of the claim]
		First of all notice that if the multi-index $\beta$ has an entry $\beta_m =1$, then the whole product under $\prod_{m=1}^{M}$ goes to $0$.
		Besides, if $\beta_m=0$, then the relative factor is $1$ and therefore does not influence the product.
		In equations:
		
		\begin{align*}
			&\prod_{m=1}^M
			\left(
			\1_{\beta_m = 0} + \1_{\beta_m \geq 2} \Iu^{\beta_m -2} 
			\right) = 
			\prod_{
				\substack{
					m=1
					\\
					\beta_m \neq 0
				}
			}^M
			\left(
			\1_{\beta_m \geq 2} \Iu^{\beta_m -2} 
			\right)
			\
			= 
			\
			\prod_{
				\substack{
					m=1
					\\
					\beta_m \neq 0
				}
			}^M
			(\1_{\beta_m \neq 1} \Iu^{\beta_m -2} )
			=
			\\
			&\qquad
			=
			\1_{\beta_m \neq 1}
			\prod_{
				\substack{
					m=1
					\\
					\beta_m \neq 0
				}
			}^M
			\Iu^{\beta_m}
			\:
			\cdot
			\:
			\prod_{
				\substack{
					m=1
					\\
					\beta_m \neq 0
				}
			}^M
			\Iu^{-2}\
			=			
			\1_{\beta_m \neq 1}
			\Iu^{\sum_{\beta_m \neq 0} \beta_m}
			\Iu^{-2\sum_{\beta_m \neq 0} 1} 
		\end{align*}
		
		Therefore:
		
		\begin{equation*}
			\begin{aligned}
				\sum_{\beta_1 + \cdots + \beta_M =2n}\ \prod_{m=1}^{M}
				\left(
				\1_{\beta_m = 0} + \1_{\beta_m \geq 2}
				\Iu^{\beta_m -2} 
				\right)
				&=
				\sum_{\beta_1 + \cdots + \beta_{M} =2n}\
				\1_{\beta_m \neq 1}
				\Iu^{\sum_{\beta_m \neq 0} \beta_m}
				\Iu^{-2\sum_{\beta_m \neq 0} 1} 
				\\
				&= 
				\sum_{\substack{
						\beta_1 + \cdots + \beta_{M} =2n
						\\
						\forall m. \beta_m \neq 1
				}}\
				\Iu^{\sum_{\beta_m \neq 0} \beta_m}
				\Iu^{-2\sum_{\beta_m \neq 0} 1} 
				\\
				&= 
				\sum_{\substack{
						\beta_1 + \cdots + \beta_{M} =2n
						\\
						\forall m. \beta_m \neq 1
				}}\
				\Iu^{2n-2(\sum_{\beta_m \neq 0} 1)}  
			\end{aligned}
		\end{equation*}
		
		We now need to take a closer look at the multi-index $\beta = (\beta_1, \ldots, \beta_{M})$ and its entries.	
		For each $\beta_m$ define the quantity
		
		\begin{equation*}
			\# \left\{ \beta_m \neq 0 \right\}  := \# \{ 
			i 
			\:
			:\
			\:
			\beta_i  \: \text{ is non-zero entry of } \: \beta
			\}
			=
			\sum_{\beta_m \neq 0} 1
		\end{equation*}
		
		Observe that we are taking into consideration only multi-indices in this set:
		
		\begin{equation*}
			\left\{
			\beta = \left( \beta_1, \ldots , \beta_{M}  \right)\
			:\
			\norm{\beta} = 2n,\
			\forall m. \beta_m \neq 1
			\right\}
		\end{equation*}
		
		Therefore, since the minimum value that we allow in an entry $\beta_m$  is $2$, the maximum amount of non-zero entries, for a multi-index $\beta = (\beta_1, \ldots, \beta_{M})$ such that $\norm{\beta} = \beta_1 + \cdots + \beta_{M} = 2n$, will be $n$ ($n$ entries of value $2$).
		Then we can write
		
		\begin{align*}
			&\sum_{\substack{
					\beta_1 + \cdots + \beta_M =2n
					\\
					\forall m. \beta_m \neq 1
			}}\
			\Iu^{2n-2(\sum_{\beta_m \neq 0} 1)} \
			=\
			\sum_{\substack{
					\beta_1 + \cdots + \beta_M =2n
					\\
					\forall m. \beta_m \neq 1
			}}\
			\Iu^{2n-2(\# \left\{ \beta_m \neq 0 \right\} )}
			=
			\\
			\qquad
			&=
			\sum_{k =1}^n\
			\sum_{\substack{
					\beta_1 + \cdots + \beta_M =2n
					\\
					\forall m. \beta_m \neq 1
					\\
					\# \left\{ \beta_m \neq 0 \right\}  = k
			}}\
			\Iu^{2n-2k}\
			=\
			\sum_{k =1}^n\
			\Iu^{2n-2k}\
			\sum_{\substack{
					\beta_1 + \cdots + \beta_M =2n
					\\
					\forall m. \beta_m \neq 1
					\\
					\# \left\{ \beta_m \neq 0 \right\}  = k
			}} 1
			\quad
			\footnotemark[1] \leq
			\Iu^{2n-2}\
			\sum_{k =1}^n\
			\sum_{\substack{
					\beta_1 + \cdots + \beta_{M} =2n
					\\
					\forall m. \beta_m \neq 1
					\\
					\# \left\{ \beta_m \neq 0 \right\}  = k
			}} 1
		\end{align*}
		\footnotetext[1]{$\Iu\geq 1$}
		
		We can notice that the last term is simply counting:
		
		\begin{equation*}
			\sum_{\substack{
					\beta_1 + \cdots + \beta_{M} =2n
					\\
					\forall m. \beta_m \neq 1
					\\
					\# \left\{ \beta_m \neq 0 \right\}  = k
			}} 1
			=
			\# \left\{
			\beta = (\beta_m)_{m=1}^{M} \ 
			:\
			\sum \beta_m = 2n, \
			\forall m. \beta_m \neq 1, \
			\#^{\widehat{0}} (\beta ) = k
			\right\}
		\end{equation*}
		
		It can be evaluated with the classical stars and bars problem, with $k -1$ bars and $n$ stars (technically there are $2n$ stars, but the condition $\beta_m \neq 1$ is equivalent to require that between each bars we have only even quantities):
		
		\begin{align*}
			&\# \left\{
			\beta = (\beta_m)_{m=1}^{M} \
			:\
			\sum \beta_m = 2n, \
			\forall m. \beta_m \neq 1, \
			\#^{\widehat{0}} (\beta ) = k
			\right\} 
			\\
			&= \# \left(
			\text{ways to choose $k$ entries in $\beta = (\beta_1, \ldots, \beta_{M})$}
			\right) \cdot
			\\
			&\qquad
			\cdot
			\# \left(
			\text{ways to arrange $k-1$ bars between $n$ stars}
			\right)
			\\
			& =
			\left(
			\begin{gathered}
				M
				\\
				k
			\end{gathered}
			\right)
			\left(
			\begin{gathered}
				n + k -1
				\\
				k - 1
			\end{gathered}
			\right)
			\\
			&=
			\frac{M \cdot (M -1)\cdots  \cdot (M -k +1)}{k!} \cdot \frac{(n +k -1) \cdot (n +k -2)\cdots \cdot (n + 1)}{(k-1)!}
			\\
			& \leq
			\left[ M \cdot (M -1)\cdots  \cdot (M -k +1)\right]
			\left[ (n +k -1) \cdot (n +k -2)\cdots \cdot (n + 1)\right]
			\\
			&= (M)_{(k)} (n +k -1)_{(k-1)}
			\\
			&\leq
			(M)^k  (n +k)^{k-1}
		\end{align*}
		
		Then, resuming, we have
		
		\begin{align*}
			\sum_{\beta_1 + \cdots + \beta_M =2n}\ \prod_{m=1}^{M}&
			\left(
			\1_{\beta_m = 0} + \1_{\beta_m \geq 2}
			\Iu^{\beta_m -2} 
			\right)\
			\leq\
			\Iu^{2n-2}\
			\sum_{k =1}^n\
			\sum_{\substack{
					\beta_1 + \cdots + \beta_{M} =2n
					\\
					\forall m. \beta_m \neq 1
					\\
					\# \left\{ \beta_m \neq 0 \right\}  = k
			}} 1
			\leq
			\\
			&\leq
			\Iu^{2n-2}\
			\sum_{k =1}^n\
			(M)^k  (n +k)^{k-1}\
			\leq\
			\Iu^{2n-2}\
			(M)^n\
			(2n)^{n-1}\
			\sum_{k =1}^n
			1
			=
			\\
			&\leq
			(2n)^n\
			(M)^n\
			\Iu^{2n-2}\
		\end{align*}
		
		And this concludes the proof of the claim.
	\end{proof}

		(2n)^n\
		N^{(\xi + \alpha-1) 2 n + 1 -2\alpha}\
		\E{
			(\hmactive_t)^n
		}\
		\\
		&=
		\const_{\attraction,\curlywedge,n}\
		\Fup_n(\const_\dens)\
		N^{(\xi + \alpha-1) 2 n + 1 -2\alpha}\
		\E{
			(\hmactive_t)^n
		}\
		\leq
	\end{align*}
	
	Now we can use the result \ref{app:Poisson-moments} to evaluate the $n$-momenta of the Poisson process $\hmactive_t \sim \Pois(\lambda N t)$, and recall that by \ref{ch5:assumption-t,lambda>1-proofs} we assume without loss of generality that the Poisson parameter is bigger than $1$:
		
	\begin{align*}
		&\leq
			\const_{\attraction,\curlywedge,n}\
			\Fup_n(\const_\dens)\
			N^{(\xi + \alpha-1) 2 n + 1 -2\alpha}\
			\Fup(n)\
			(Nt\lambda)^n
		\\
		&=
		\const_{\attraction,\curlywedge,n,\lambda}\
		\Fup_n(\const_\dens)\
		t^n
			N^{(\xi + \alpha-1) 2 n + 1 -2\alpha}\
			N^n\
		\\
		&\leq
			\const_{\attraction,\curlywedge,n,\lambda}\
			\Fup_n(\const_\dens)\
			\Fup_n(t)\
			N^{ [2(\xi + \alpha) - 2 + 1] n + 1 -2\alpha}\
		\leq
	\end{align*}
	
	Then, for every $K = \const_\xi$ such that $0 < \const_\xi \leq 1 - 2(\alpha + \xi)$ (it is guaranteed to exist because $\alpha + \xi < \frac{1}{2}$), we have
	
	\begin{equation*}
		\leq\
			\const_{\attraction,\curlywedge,n,\lambda}\
			\Fup_n(\const_\dens)\
			\Fup_n(t)\
			N^{ -\const_\xi n + 1 -2\cdot 0}\
		=\
			\const_{\attraction,\curlywedge,n,\lambda}\
			\Fup_n(\const_\dens)\
			\Fup_n(t)\
			N^{ -\const_\xi n + 1}
	\end{equation*}
	
	And thus the proof of the proposition is finished.


\section{The second part of the law of large numbers}

To prove the second part of the law of large numbers it will be necessary to prove that a specific \rv has finite $q$-moments, but in order to better drive the comprehension and the attention of the reader this will be done after this proposition.

\begin{proposition}[The second part of the law of large numbers]\label{ch5:prop:LLN2}

\end{proposition}

	\end{proof}

\section{Composing all the pieces together}

\begin{proposition}\label{ch5:LLN}[Law of large numbers]
	
	Let $\drift(t, \cdot)$ be the generalized drift of parameter $0 \leq \alpha < \frac{1}{2}$ and constant $\const_\curlywedge > 0$ and let $\mfsol_t^N$ be the solution to the mean-field system with $N$ amoebas, as defined in \ref{ch5:def:generalized-attr&drift}.
	Let $t\leq T$ and let $\xi >0$ be such that $0 < \alpha + \xi <\frac{1}{2}$.
	Choose a value $0 < \const_\xi \leq 1 - 2(\alpha + \xi)$.
	
	Then it is valid the law of the large numbers, \ie there exists $\overline{N} = \overline{N}(\alpha, \const_\curlywedge) > 0$ such that for all $n \in \N$ exists an increasing positive continuous function $\Fup_n$ (independent from $N, \alpha, \xi$) so that for all $N > \overline{N}$
	\begin{equation}
		\PrbOf{
			\Norm{
				\drift^N(t, \mfsol_t^N)
				-
				\mfdrift^N(t, \mfsol_t^N)
			}_\infty
			\geq
			2N^{-\xi}
		}
		\leq
		\Fup_n(T)\
		N^{-\const_\xi n + 1}
	\end{equation}
	
	In particular, if $\drift(t, \cdot)$ is the usual drift of \ref{ch2:def:drift-&-diffusion}, then $\alpha$ can be chosen equal to $0$.
\end{proposition}

\begin{proof}
	Once again we make the assumptions of remark \ref{ch5:assumption-t,lambda>1-proofs}, \ie that $t,\lambda\geq1$.
	Recall that from \ref{ch5:sketch-dividi-et-impera} we want to use the strategy "dividi et impera".
