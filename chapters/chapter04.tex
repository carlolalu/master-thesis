\section{Dini derivativation and derivation of $J_t^N$}

In the main result we will need to consider the Dini derivation of $\sup_{0 \leq t \leq T} \norm{\sol_t^N-\mfsol_t^N}_\infty$.

\subsection{Dini derivatives}

We follow \autocite{dini-derivatives} and represent the work of \autocite{ana}.

\begin{definition}[Dini derivatives]
	For any function $f: \R \longrightarrow \R$ we define the \emph{right upper, right lower, left upper \emph{and} left lower Dini derivatives} in $t$ as:
	\begin{align*}
		& \diniu^+ f (t) = \limsup_{h\rightarrow 0^+} \frac{f(t+h) - f(t)}{h}
		& \dinil^+ f (t) = \liminf_{h\rightarrow 0^+} \frac{f(t+h) - f(t)}{h}
		\\
		& \diniu^- f (t) = \limsup_{h\rightarrow 0^-} \frac{f(t+h) - f(t)}{h}
		& \dinil^- f (t) = \liminf_{h\rightarrow 0^-} \frac{f(t+h) - f(t)}{h}		
	\end{align*}
	For any function $g: \R^\Ch \longrightarrow \R$ and any direction (\ie unit vector) $v\in \R^\Ch$ we define the \emph{upper \emph{and} lower dini derivative in the direction $v$} as:
	\begin{align*}
		& \diniu_v g (x) = \limsup_{h\rightarrow 0^+} \frac{g(x+hv) - g(x)}{h}
		& \dinil_v g (x) = \liminf_{h\rightarrow 0^+} \frac{g(x+hv) - g(x)}{h}
	\end{align*}
\end{definition}

\begin{remark}
	The right upper Dini derivative $\diniu^+ f (t)$ might be signed also as $\diniu^+_t f (t)$, and analogous for the others.
	The Dini derivative of the multivariate case has been added for the sake of completeness, but it will never be used in this work.
\end{remark}

\begin{remark}
	Notice that the Dini derivatives are always defined if we allow them to take also values in $[-\infty, \infty]$.
	Furthermore it is easy to see that
	\begin{itemize}
		\item If $f$ is locally Lipschitz, then its dini derivatives are all finite.
		\item If $f$ is right-differentiable $\partial^+ f (t) = \diniu^+ f (t) = \dinil^+ f (t)$.
			Analogously happens if $f$ is left-differentiable.
		\item If $f$ is differentiable then $\partial f (t) = \partial^+ f (t) = \partial^- f (t)$ and all the dini derivatives coincide.
	\end{itemize}
\end{remark}

\begin{lemma}[Properties of Dini derivatives]
	Let $f,h: \R \longrightarrow \R$ be $\Cspace^0$.
	Then
	\begin{itemize}
		\item $\diniu^+ (f + h) \leq \diniu^+ f + \diniu^+h$
		\item If $f > 0$. then $\diniu^+ (fh) \leq (\diniu^+f) h + f(\diniu^+h)$
		\item (\autocite[Theorem 11]{dini-derivatives}) If $f$ has \emph{finite} Dini derivative $\diniu^+$ at every point in $t\in\R$, and if $\diniu^+f$ is Lebesgue integrable on $[a,b]$, then
			\begin{equation}
				f(b) - f(a) = \int_a^b \diniu^+ f (t) dt.
			\end{equation}
	\end{itemize}
\end{lemma}

The proof of the first two properties is easily achieved exploiting the properties of $\limsup$, while the third one is found in the quoted article.

With this background in mind, we state the next result, of which the proof will be achieved by the end of the section (\ref{ch4:proof:del-dini-of-all}).

\begin{lemma}\label{ch4:lemma:del-dini-of-all}
	Let $\sol^N$ be the solution to the model \ref{ch2:model} and $\mfsol^N$ its coupled mean-field particles.
	Let $\drift$ and $\mfdrift$ be the correspondent drifts for the two.
	Then we have that
	\begin{equation}
		\diniu_T^+ \sup_{0 \leq t \leq T} \norm{\sol_t^N-\mfsol_t^N}_\infty
		\leq
		 d N \sup_{0 \leq t \leq T} \norm{\drift^N(t, \sol_t^N) - \mfdrift(t, \mfsol_t^N)}_\infty
	\end{equation}
	where $\NumParticles_T = N + \hmactive_T$ is the number of particles present at time $T$ in the whole system.
\end{lemma}

\subsection{Advance beyond the $\sup$}

In this part we want to show that

\begin{equation}\label{ch4:experiment-label}
\diniu_T^+ \sup_{0 \leq t \leq T} \norm{\sol_t^N-\mfsol_t^N}_\infty
\leq
\sup_{0 \leq t \leq T} \diniu^+_t \norm{\sol_t^N-\mfsol_t^N}_\infty
\end{equation}

Consider the random function

\begin{equation}
\Ch(t) := \norm{\sol_t^N - \mfsol_t^N}_\infty
\end{equation}

and look at its paths.
Such function is surely continuous between birth times $\btime_m$, but what happens at such instants?

\begin{lemma}
The paths of $\Ch(t)$ are \cadlag and have only a finite number of discontinuouties on $[0,T]$.
Furthermore, considering $\Ch(t-) := \lim_{h\rightarrow 0^-} \Ch(t+h)$, we have that
\begin{equation}
\forall \btime_m. \Ch(\btime_m-) \leq \Ch(\btime_m)
\end{equation}
\end{lemma}

\begin{remark}
Notice that this does \emph{not} necessarily mean that $\Ch$ has increasing paths, it only means that such paths, if they do have a jump, it's necessarily a jump up.
Such condition, on \cadlag functions $g$, can be simply stated as:
\begin{equation}
	\forall t. \quad g(t-) \leq g(t+)
	\quad 
	\left( =g(t) \right)
\end{equation}
\end{remark}

\begin{proof}
Since both $\sol_t^N$  and $\mfsol_t^N$ are solutions to SDEs on $I_m \smallsetminus \{ \btime_{m+1} \} = [\btime_m, \btime_{m+1})$, they are continuous on such interval.
This, together with the fact that $\Norm{\cdot}_\infty$ is $\Cspace^0$ and that $\forall \omega$ the number of birth times $\btime_m$ in $[0,T]$ is finite, means that the discontinuouties n $[0,T]$ is finite.
Furthermore we have that $[\btime_m, \btime_{m+1})$ is closed on the left, thus the paths are right-continuous.
By definition we have also that 
\begin{align}
	& \lim_{t\rightarrow \btime_{m+1}} \sol_t^N = (\amba_{\btime_{m+1}}^1,\ldots,\amba_{\btime_{m+1}}^N, \camp_{\btime_{m+1}}^1, \ldots, \camp_{\btime_{m+1}}^m )
\end{align}
and analogously for $\mfsol_t^N$. Therefore the paths of $\norm{\sol_t^N - \mfsol_t^N}_\infty$ are \cadlag.

Let us now prove that  $\Ch(\btime_m-) \leq \Ch(\btime_m)$ for every discontinuouty $\btime_m$ in $[0,T]$.
We have that
\begin{align*}
	\Ch(\btime_m-) &= \lim_{t\rightarrow\btime_{m+1}^-} \Ch(t) = \lim_{t\rightarrow\btime_{m+1}^-} \norm{ \sol_t^N - \mfsol_t^N}_\infty
	\\
	& = \lim_{t\rightarrow\btime_{m+1}^-} \max \left\{ 
		\max_{p \in \{ 1, \ldots, N\}} \norm{
			\amba_t^p - \mfamba_t^p
			},
		\max_{1\leq i \leq m } \norm{
			\camp_t^i - \mfcamp_t^i
			}
		\right\}
	\\
	& = \max \left\{ 
	\max_{p \in \{ 1, \ldots, N\}} \norm{
		\amba_{\btime_{m+1}}^p - \mfamba_{\btime_{m+1}}^p
	},
	\max_{1\leq i \leq m } \norm{
		\camp_{\btime_{m+1}}^i - \mfcamp_{\btime_{m+1}}^i
	}
	\right\}
	\\
	& \leq \max \left\{ 
	\max_{p \in \{ 1, \ldots, N\}} \norm{
		\amba_{\btime_{m+1}}^p - \mfamba_{\btime_{m+1}}^p
	},
	\max_{1\leq i \leq m } \norm{
		\camp_{\btime_{m+1}}^i - \mfcamp_{\btime_{m+1}}^i
	},
	\norm{
		\camp_{\btime_{m+1}}^{m+1} - \mfcamp_{\btime_{m+1}}^{m+1}
	}
	\right\}
	\\
	& = \norm{ \sol_{\btime_{m+1}}^N -\mfsol_{\btime_{m+1}}^N}_\infty = \Ch(\btime_{m+1})
\end{align*}
\end{proof}

\begin{remark}
	Notice that, as noted in \ref{ch3:rmk:mf-not-c0}, we have that the trajectory of the mean-field cAMP does not start on any amoeba, \ie there is no bifurcation of paths at birth times, hence the possibility of a jump at each birth time $\btime_m$.
\end{remark}

\begin{definition}
Let $\SkoroF$ be the set of functions $g: \R_+ \longrightarrow \R_+$ such that:
\begin{itemize}
\item $g(0)=0$
\item $g$ is \cadlag
\item  For each $T>0$, $g$ has finite amount of discontinuouity in $[0,T]$
\end{itemize}
Besides we define the set $\SkoroFUp $ as the subset of elements $g \in \SkoroF$ such that if $g$ is discontinuous in $t_i$ then $g(t_i-) < g(t_i)$, or equivalently, such that $\forall t. g(t-) \leq g(t+)$.
\end{definition}

By the previous lemma the paths of $\Ch$ are elements in $\SkoroFUp$.

\begin{lemma}
Let $g \in \SkoroFUp$ and consider $g_\star (t) := \sup_{0\leq s \leq t} g(s)$.
Then
\begin{equation}
g_\star (t) = \max_{0\leq s \leq t} g(s)
\end{equation}
and $g_\star \in \SkoroFUp$.
\end{lemma}

\begin{proof}
	Let $\{\tau_1, \ldots, \tau_m\}$ be the set of discontinuouities of $g$ in $[0,t]$, and define $\tau_0:=0, \tau_{m+1} := t$.
	Define then, for each $i\in\{1,\ldots, m\}$, the function
	\begin{equation*}
		g_i (s) := \1_{[\tau_{i-1}, \tau_i)}(s) g(s) + \1_{\{ \tau_i\}}(s) g(\tau_i-)
		=
		\left\{ 
		\begin{aligned}
			&g(s) 	& s \in [\tau_{i-1}, \tau_i)
			\\
			&g(\tau_i - ) 	& s = \tau_i
		\end{aligned}
		\right.
	\end{equation*}
	The functions $g_i$ are obviously continuous, and since $[\tau_{i-1}, \tau_i]$ is a closed interval, we have that $\sup_s g_i(s) = \max_s g_i(s)$.
	We have two possible cases: either $\tau_i$ is a maximal point for $g_i$, or it is not.
	If $\tau_i$ is a maximal point for it we have that 
	
	\begin{equation}
		g_i (s) \leq g_i (\tau_i) = g(\tau_i -) \leq g(\tau_i)
		\quad \Rightarrow 	\max g_i \leq g (\tau_i)
	\end{equation}
	
	Otherwise if $\tau_i$ is not a maximal point for $g_i$, we have that $\exists s_i \in [\tau_{i-1}, \tau_i)$ such that $\forall s.\; g_i (s) \leq g(s_i) = g_i (s_i) $.
	Since $g_i =g$ on all points but on $\tau_i$, and since $(g_i)_i$ is a sequence with a finite number of elements, we  can affirm that in all cases
	\begin{equation}
		\sup_{s \in [\tau_{i-1}, \tau_i]} g(s) = \max \left\{ \max_{s \in [\tau_{i-1}, \tau_i]} g_i(s), g(\tau_i) \right\} = \max_{s \in [\tau_{i-1}, \tau_i]} g(s)
	\end{equation}
	
	Hence it is obvious that
	
	\begin{align}
		\sup_{s \in [0,t]} g(s) 
		& = \max_i \left\{ \sup_{s \in [\tau_{i-1}, \tau_i]} g(s) \right\} = \max_i \left\{ \max_{s \in [\tau_{i-1}, \tau_i]} g(s) \right\}
		\\
		& = \max_{s \in [0,t]} g(s) 
	\end{align}
	
	Therefore we have that $g_\star(t) = \sup_{[0,t]} g = \max_{[0,t]} g$.
	
	Let us now prove that $g_\star$ is an element of $\SkoroFUp$.
	\begin{claim}
		The set of discontinuouities of $g_\star$ is contained in $\{\tau_1, \ldots, \tau_m\}$, the set of discontinuouities of $g$.
	\end{claim}
	
	\begin{figure}[h]
		\begin{tikzpicture}
		\end{tikzpicture}
		\caption{[[[HERE DRAW TIKZ FIGURE of g and g star]]]}
	\end{figure}
	\label{ch4:fig:sup-of-cadlag-functions}
	
	\begin{proof}
		Let $t$ be a point where $g$ is continuous.
		We show that also $g_\star$ is continuous in $t$.
		
		We have two possibilities: either $g(t) = g_\star(t)$, or $g(t) < g_\star (t)$.
		\begin{itemize}
			\item If $g(t) < g_\star (t)$, since $g$ is $\Cspace^0$ in $t$, there is a neighbourhood $U_t$ of $t$ such that $g|_{U_t} < g_\star(t)$, and since $g_\star(t) = \max_{[0,t]} g$, we have that $(g_\star)|_{U_t} \equiv g_\star(t)$, and thus that $g_\star$ is continuous in $t$.
			\item If $g(t) = g_\star(t)$, consider an $\epsilon>0$ small enough so that $g$ is monotonous on the left and right intervals $U_\epsilon^-:= (t-\epsilon, t)$ and $U_\epsilon^+:=(t,t+\epsilon)$. 
			This can be done otherwise the function $g$ would oscillate infinite times in a neighbourhood, and we could construct a sequence $(t_n)_n$ such that $(g(t_n))_n$ does not converge to $g(t)$, contraddicting the continuouity in $t$.
			Let us show that $g_\star$ is continuous on both left and right neighbourhoods, being thus continuous.
			\begin{itemize}
				\item[$\big( U_\epsilon^- \big)$] $g$ can be either constant or strictly increasing on $U_\epsilon^-$ (otherwise if $g$ would be strictly decreasing, we would have a point $s<t$ such that $g(s) > g(t) = g_\star(t) = \max_{[0,t]} g$, contraddiction).
				If $g$ is constant on this left neighbourhood, then so is $g_\star$ and thus it is left continuous.
				If $g$ is stricly increasing on $U_\epsilon^-$ we have either that the two functions coincide on such domain, or that $g_\star$ is constant on it, and in both case we have that $g_\star$ is left continuous.
				\item[$\big( U_\epsilon^+ \big)$] Consider now the case on $U_\epsilon^+=(t,t+\epsilon)$: if $g$ is non-increasing on it, then $(g_\star)|_{U_\epsilon^+} \equiv g_\star (t)$.
				If instead $g$ is increasing on it, the two functions coincide on it.
				In either case, $g_\star$ is right-$\Cspace^0$.
			\end{itemize}
		\end{itemize}
		
		Therefore $g_\star$ is continuous on $t$.
	\end{proof}
	
	\begin{claim}
		$g_\star$ is right-$\Cspace^0$, \ie $g_\star(t) = g_\star(t+)$.
	\end{claim}
	\begin{proof}[Proof of claim]
		\begin{equation*}
		\begin{aligned}
			g_\star(t+) 
			&= \lim_{h \rightarrow 0^+} g_\star(t+h) = \lim_{h \rightarrow 0^+} \sup_{s \leq t+h} g_\star(s)
			\\
			&= \lim_{h \rightarrow 0^+} \max \Big\{ 
				\underbrace{\sup_{s\in [0,t)} g(s)}_{\leq \sup_{s\in [0,t]} g(s) = g_\star(t)}
				,\quad
				\underbrace{\sup_{s \in [t, t+h]} g(s) }_{
					\sup_{\delta \in [0, h]} g(t+\delta) 
					}
			\Big\}			
			\\
			& \leq \max \{ g_\star (t), \lim_{h\rightarrow 0^+} \sup_{\delta \in [0, h]} g(t+\delta) \}
			\\
			& = \max \{ g_\star (t), \limsup_{s\rightarrow t^+} g(s) \}
			& = \max \{ g_\star (t), g(t) \} = g_\star(t)
		\end{aligned}
		\end{equation*}
		Thus $g_\star(t+) \leq g_\star(t)$, but since $g_\star$ is increasing by definition we also have that $g_\star(t+) \geq g_\star(t)$, therefore $g_\star(t+) = g_\star(t)$.
	\end{proof}
	\begin{claim}
		For each $t$ we have that $\exists g_\star(t-)$ and $g_\star(t-) \leq g_\star(t)$.
	\end{claim}
	\begin{proof}[Proof of claim]
		If $g_\star$ is $\Cspace^0$ in $t$ the claim is obvious, let us thus assume that $g_\star$ is discontinuous in $t$.
		Notice that since $g_\star$ is right-continuous and increasing, if the left limit $g_\star (t-)$ exists, then it must necessarily be that $g_\star(t-) < g_\star(t)$.
		Therefore we need only to show that such left limit exists.
		
		Let $(t_n)_n$ be a sequence of points such that $\forall n. \: t_n \leq t_{n+1} <t$ and $t_n \longrightarrow t$.
		Since the set of discontinuouities of $g$ on $[0,T]$ is finite, then it is discrete, thus there is an $\epsilon>0$ such that $g$ (and hence $g_\star$) are $\Cspace^0$ on $U_\epsilon^- := (t-\epsilon, t)$ and on $U_\epsilon^+:= (t,t+\epsilon)$.
		We then observe, that since $(t_n)$ is a non-decreasing sequence and $g_\star$ is a non-decreasing function, we have that 
		\begin{equation}
			g_\star (t_n) \leq g_\star ( t_{n+1} )
		\end{equation}
		\ie that the sequence $( g_\star (t_n) )_n$ is non-decreasing.
		Furthermore we have that $( g_\star (t_n) )_n$ is bounded by $g_\star(t)$ (because of the definition of $g_\star$).
		Then, being the sequence $( g_\star (t_n) )_n$ bounded and non-decreasing, it has a limit $l$.
		But being such sequence arbitrary, we have that
		\begin{equation}
			l = \lim_n g_\star (t_n) = \lim_{s\rightarrow t^-} g(s) = g(t-)
		\end{equation}
	\end{proof}
	And thus $g_\star \in \SkoroFUp$ and the proof is concluded.
\end{proof}

\begin{lemma}
Let $g$ be a function in $\SkoroFUp$.
Then
\begin{equation}
\diniu^+_t g_\star(t)
=
\diniu^+_t \sup_{0\leq s \leq t} g(s) 
\leq
\sup_{0\leq s \leq t} \diniu^+_s g(s)
=
\left( \diniu^+ g \right)_\star (t)
\end{equation}\label{ch4:eq:sup-dini-exchange}
\end{lemma}

\begin{proof}
Observe that since $g \geq 0$ and since $g_\star$ is non-decreasing, we have that

\begin{equation}
	\lhs (\ref{ch4:eq:sup-dini-exchange}) = \lim_{h\rightarrow 0^+} \sup_{\delta \in (0,h]} \frac{g_\star(t+\delta) - g_\star(t)}{\delta} \geq 0
\end{equation}
Thus we have only two possibilities:
\begin{equation}
	\lhs (\ref{ch4:eq:sup-dini-exchange}) = 0 
	\qquad \text{or} \qquad
	\lhs (\ref{ch4:eq:sup-dini-exchange}) > 0 
\end{equation}

Consider the case $\lhs (\ref{ch4:eq:sup-dini-exchange}) = 0$.
Then it is sufficient to show that $\rhs (\ref{ch4:eq:sup-dini-exchange}) \geq 0$ to obtain the result.
Since $g \geq 0$, we trivially have that $g(t) \geq g(0)$, and thus

\begin{equation}
	\forall \delta >0 \quad \frac{g(\delta) - g(0)}{\delta} \geq 0
	\qquad
	\diniu_0^+g(0) \geq 0	
\end{equation}

And thus in such case we have that

\begin{equation}
	\lhs (\ref{ch4:eq:sup-dini-exchange}) = 0 
	\leq
	\diniu_0^+g(0)
	\leq
	\sup_{s \in [0,t]} \diniu_s^+g(s)
	=
	\rhs (\ref{ch4:eq:sup-dini-exchange})
\end{equation}

And so this case is covered.

Let us then assume that $\lhs (\ref{ch4:eq:sup-dini-exchange}) > 0 $.
We have that

\begin{equation}
\begin{aligned}
	\lhs (\ref{ch4:eq:sup-dini-exchange}) > 0 
	& \Longleftrightarrow 
		\forall h \geq 0. \exists 0< \delta \leq h. 
		\quad
		\frac{g(t + \delta) - g(t)}{\delta} > 0
	\\
	& \Longleftrightarrow 
		\forall h \geq 0. \exists 0< \delta \leq h. 
		\quad
		g(t + \delta) - g(t)> 0
	\\
	& \Longleftrightarrow 
		\forall n \in \N. \exists 0< \delta_n \leq \frac{1}{n}. 
		\quad
		g(t + \delta_n) - g(t)> 0
		\label{ch4:cond:proof-dini-T1}
\end{aligned}
\end{equation}

The strategy is to show that under the condition \ref{ch4:cond:proof-dini-T1} in $t$, we can approximate the Dini derivative of $g_\star$ with the one of $g$, obtaining thus:

\begin{equation*}
\lhs = \diniu_t^+g_\star(t)
\simeq \diniu_t^+g(t)
\leq \sup_{s\in [0,t]} \diniu_s^+g(s)
= \rhs
\end{equation*}

This will require some preliminary claims to be proved.

\begin{claim}\label{ch4:claim:g=gstar}
	Under the condition \ref{ch4:cond:proof-dini-T1} we have that $g(t) = g_\star(t)$.
\end{claim}
\begin{proof}
	By \ref{ch4:cond:proof-dini-T1} we have that $\forall n. \exists 0 < \delta_n \leq \frac{1}{n}$ such that
	\begin{equation}
		\max_{[0, t + \delta_n]} g = g_\star (t + \delta_n) 
		>^{\ref{ch4:cond:proof-dini-T1}}
		g_\star (t) = \max_{[0, t]} g
	\end{equation}
	Then $\forall n . \exists r_n \in (0, \delta_n] \subseteq (0, \frac{1}{n})$ such that $g(t + r_n ) = g_\star (t + \delta_n) > g_\star (t)$.
	This, together with the fact that $r_n \xrightarrow{n} 0$ and that the functions involved are right-continuous, let us infer that
	\begin{equation}
		g(t) \leq g_\star (t) \leq \lim_n g(t + r_n) = g(t+) = g(t)
	\end{equation}
	And then $g(t) = g_\star (t)$.
\end{proof}

\begin{claim}\label{ch4:claim:exploit-xi-of-delta}
	Under the condition \ref{ch4:cond:proof-dini-T1} in $t$ consider $\delta >0$ such that $g,g_\star$ are $\Cspace^0$ on $U_\delta^+:= [t,t+\delta)$.
	Define
	\begin{equation}
		\xi_{t,\delta} := \xi_\delta := \xi := \inf \left\{ \nu \geq 0 \ | \  g_\star(t + \nu) = g_\star (t + \delta) \right\}
	\end{equation}
	Then $\xi$ is strictly positive, is $\leq \delta$ and is such that $g(t+\xi) = g_\star (t+ \xi) = g_\star(t + \delta)$.
\end{claim}

\begin{proof} We have that $\xi \leq \delta$ is trivially true by definition.

	\begin{align*}
		A :&= \left\{ \nu \geq 0 \ | \  g_\star(t + \nu) = g_\star (t + \delta) \right\}

	Hence $\xi > 0$.
	
	\paragraph{$(\invquest g(t+\xi) = g_\star (t+ \xi) ?)$} Suppose ad absurdum that $g(t+\xi) < g_\star (t + \xi)$.

	Then $\exists 0 < \nu < \xi$ such that
	\begin{equation}
		g(t + \nu) = g_\star (t+ \xi) = \max_{[0, t+ \nu]} g = g_\star (t + \nu)
	\end{equation}
	And this is a contraddiction, because $\nu < \xi = \min A$, but $\nu \in A$.
	
	Finally, it is true by definition that $g_\star (t+ \xi)  = g_\star (t+ \delta)$.
\end{proof}
Now we have finally all the instrument to prove the final result.

	\begin{align*}
	\lhs(\ref{ch4:eq:sup-dini-exchange}) 
	&= \diniu^+_t g_\star(t) 
	\\
	&= \lim_{h \rightarrow 0^+} \sup_{\delta \in (0, h]} \frac{
		g_\star(t + \delta) - g_\star (t)
		}{\delta}
	\\
	&= \lim_{h \rightarrow 0^+} \sup_{\delta \in (0, h]} \frac{
		g_\star(t + \delta) - g (t)
	}{\delta}
	&& \text{by }\ref{ch4:claim:g=gstar}
	\\
	&= \lim_{h \rightarrow 0^+} \sup_{\delta \in (0, h]} \frac{
		g(t + \xi_\delta) - g (t)
	}{\delta}
	&& \text{by }\ref{ch4:claim:exploit-xi-of-delta}
	\end{align*}
	
Then, since $0 < \xi_\delta \leq \delta$ (by \ref{ch4:claim:exploit-xi-of-delta}), we have that $\frac{1}{\delta} \leq \frac{1}{\xi_\delta}$ and thus 
	\begin{align*}
	&\leq \lim_{h \rightarrow 0^+} \sup_{\delta \in (0, h]} \frac{
			g(t + \xi_\delta) - g (t)
		}{\xi_\delta}
		&& \text{by }\ref{ch4:claim:exploit-xi-of-delta}
	\\
		&\leq \lim_{h \rightarrow 0^+} \sup_{\delta \in (0, h]} \sup_{\nu \in (0, \delta]}
			\frac{
			g(t + \nu) - g (t)
			}{\nu}
	\\
	& = \lim_{h \rightarrow 0^+} \sup_{\delta \in (0, h]}
		\frac{
			g(t + \delta) - g (t)
		}{\delta}
	\\
	&= \diniu^+_t g(t) 
	\leq
	\sup_{s\in [0,t]} \diniu^+_s g(s) = \rhs(\ref{ch4:eq:sup-dini-exchange})
	\end{align*}
	And thus the result stated above is valid.
\end{proof}

\begin{remark}
	One might be tempted to ask why we did not exploit the stronger result 
	\begin{equation}\label{ch4:remark:del-dini-stronger?}
		\diniu^+_t g_\star(t) 
		\leq
		\diniu^+_t g(t)
	\end{equation}
	for the statement.
	But after observing more carefully one finds out that such result is valid only under the condition \ref{ch4:cond:proof-dini-T1}, and not in general.
	A quick check is given by the fact that $g$ in general is \emph{not} increasing and thus $\diniu^+_t g(t)$ might be negative, while $g_\star$ is non -decreasing, hence $\diniu^+_t g_\star(t)$ is always non-negative, which would break the statement \ref{ch4:remark:del-dini-stronger?}.
\end{remark}

\subsection{Get rid of the integral sign in the derivation}

Notice that since the mean-field particles are coupled with the system \ref{ch2:model}, and since the initial conditions $\adens_0$ are the same for all the amoebas, we have that

\begin{align*}
	\norm{\sol_t^N-\mfsol_t^N}_\infty 
	&= \norm{
		\sol_0^N 
		+ \int_0^t \drift(s, \sol_s^N) d s
		+ 
		\sigma B_t^N
	-
	\left(
		\mfsol_0^N 
		+ \int_0^t \drift(s, \mfsol_s^N) d s
		+ 
		\sigma B_t^N
	\right)
	}_\infty
	\\
	&=
	\norm{
		\int_0^t \drift(s, \sol_s^N) - \mfdrift(s, \mfsol_s^N) d s
	}_\infty
\end{align*}

Therefore, if we try to Dini-derive such quantity we might sketch this:

\begin{align*}
	\diniu^+_t 
	\;
	\norm{\sol_t^N-\mfsol_t^N}_\infty 
	&=
		\diniu^+_t 
		\;
		\Big|
\underbrace{ \int_0^t \drift(s, \sol_s^N) - \mfdrift(s, \mfsol_s^N) d s}_{:= I(t)}
\Big|_\infty
	\\
	& \quad \left( \text{if $\forall j.\; I_j(t)$ is right-differentiable} \right)
	\\
	& =
		\partial^+_t
		\;
		\norm{
			\int_0^t \drift(s, \sol_s^N) - \mfdrift(s, \mfsol_s^N) d s
		}_\infty
	\\
	& =
		\underbrace{
			\left[ \nabla_x \norm{x}_\infty \right]_{x = I(t)}
		}_{\invquest \leq (1, \ldots, 1) ?}
		\Big(
			\underbrace{
				\partial^+_t
				\int_0^t \drift_i(s, \sol_s^N) - \mfdrift_i(s, \mfsol_s^N) d s
			}_{\invquest = \drift_i(t, \sol_t^N) - \mfdrift_i(t, \mfsol_t^N) ?}
		\Big)_{i=1}^{d\NumParticles_t}
	\\
	& \leq \sum_i \drift_i(t, \sol_t^N) - \mfdrift_i(t, \mfsol_t^N)
	\\
	& \leq \sum_i \norm{\drift_i(t, \sol_t^N) - \mfdrift_i(t, \mfsol_t^N)} = \norm{\drift(t, \sol_t^N) - \mfdrift(t, \mfsol_t^N)}_1
	\\
	& \leq d\NumParticles_t \norm{\drift(t, \sol_t^N) - \mfdrift(t, \mfsol_t^N)}_\infty
\end{align*}

To prove that this sketch is yielding the correct result we need to analyse and confirm a couple of results on the way.
The first is an easy computation.

\begin{lemma}\label{ch4:lemma:derive-infinity-norm}
	Considering eventually the derivative $\partial_i \norm{x}_\infty$ in the distributional sense we have that
	\begin{equation}
		\partial_i \norm{x}_\infty = sgn(x_i)\1_{x_i = max_j \norm{x_j}} \leq 1
	\end{equation}
	and therefore that $\nabla \norm{x}_\infty \leq (1, \ldots, 1)$.
\end{lemma}
\begin{proof}
	Let us suppose that $\norm{x}_\infty = x_i$.
	Then
	\begin{align*}
		\partial_i \norm{x}_\infty = \partial_i |x_i| = sgn(x_i)
	\end{align*}
	If instead there is $j \neq i$ such that $\norm{x}_\infty = x_j$, then
	\begin{align*}
		\partial_i \norm{x}_\infty = \partial_i |x_j| = 0
	\end{align*}
	Hence $\partial_i \norm{x}_\infty = sgn(x_i)\1_{x_i = max_j \norm{x_j}} \leq 1$.
\end{proof}

The second one is a confirmation of an intuitive fact:

\begin{lemma}\label{ch4:lemma:right-diff-of-integral}
	The quantity 
	\begin{equation}
		\int_0^t \drift(s, \sol_s^N) - \mfdrift(s, \mfsol_s^N) d s
	\end{equation}
	is right-differentiable in each of its coordinates and is such that
	\begin{equation}
		\partial^+ \int_0^t \drift_i(s, \sol_s^N) - \mfdrift_i(s, \mfsol_s^N) d s = \drift_i(t, \sol_t^N) - \mfdrift_i(t, \mfsol_t^N)
	\end{equation}
	almost everywhere.
\end{lemma}

\begin{proof}
	Consider
	\begin{equation}
		\Zhi_i(t) := \int_0^t \drift_i(s, \sol_s^N) - \mfdrift_i(s, \mfsol_s^N) d s,
	\end{equation}
	the $i$-th coordinate of the object of interest.
	Then our thesis is that for all $t$
	\begin{align*}
		\exists \quad \lim_{h\rightarrow 0^+} \frac{\Zhi_i(t+h) - \Zhi_i(t)}{h} 
		&= \lim_{h\rightarrow 0^+} \frac{1}{h} \int_{t}^{t+h} \drift_i(s, \sol_s^N) - \mfdrift_i(s, \mfsol_s^N) d s
		\\
		&= \lim_{h\rightarrow 0^+} \frac{1}{
				\lebsge((t, t+ h))
			} \int_{(t,t+h)} \drift_i(s, \sol_s^N) - \mfdrift_i(s, \mfsol_s^N) d\lebsge(s)
		\\
		&= \drift_i(t, \sol_t^N) - \mfdrift_i(t, \mfsol_t^N)
	\end{align*}
	To prove this we just need to exploit the Lebesgue differentiation theorem (see \autocite[Theorem 3.21]{folland}), \ie check that the integrand is in $\Lpspace^1_{loc}$ (true by the definition of the solutions $\sol$ and $\mfsol$) and that $\{E_h\}_{h>0} := \big\{ (t,t+h) \big\}_{h>0}$ is a collection of Borel sets shrinking nicely to $t$.
	The latter condition means to verify that:
	\begin{itemize}
		\item $\forall h. \; E_h \subseteq B_t(h)$, the ball of radius $h$ centered in $t$.
		\item $\exists \alpha >0$ such that $\forall h. \; \lebsge (E_h) > \alpha \lebsge (B_t(h))$.
	\end{itemize}
	
		\begin{figure}[h]
		\begin{tikzpicture}
		\end{tikzpicture}
		\caption{[[[HERE DRAW TIKZ FIGURE]]]}
		\end{figure}
	
	The verification is trivial: for each $h>0$ we have that $E_h = (t,t+h) \subseteq (t-h, t+h) = B_t(h)$ and that 
	\begin{equation}
		\lebsge(E_h) = h = \frac{1}{2} \cdot 2h = \frac{1}{2} \lebsge(B_t(h)) > \frac{1}{3} \lebsge(B_t(h)).
	\end{equation}
	
	Then the Lebesgue differentiation theorem guarantees us that
	
	\begin{align*}
		\lim_{h\rightarrow 0^+} \frac{1}{h} \int_{t}^{t+h} \drift_i(s, \sol_s^N) - \mfdrift_i(s, \mfsol_s^N) d s
		&= \drift_i(t, \sol_t^N) - \mfdrift_i(t, \mfsol_t^N)
		\\
		&= \lim_{h\rightarrow 0^+} \frac{1}{
			\lebsge(E_h)
		} \int_{E_h} \drift_i(s, \sol_s^N) - \mfdrift_i(s, \mfsol_s^N) d\lebsge(s)
		\\
		&= \drift_i(t, \sol_t^N) - \mfdrift_i(t, \mfsol_t^N)
	\end{align*}
	almost everywhere, and this concludes the proof.
\end{proof}

Finally we can show that the result \ref{ch4:lemma:del-dini-of-all}:

\begin{proof}[Proof of \ref{ch4:lemma:del-dini-of-all}]\label{ch4:proof:del-dini-of-all}
	In \ref{ch4:result:dini-of-sup} we have already shown that the inequality
		\begin{equation}
		\diniu_T^+ \sup_{0 \leq t \leq T} \norm{\sol_t^N-\mfsol_t^N}_\infty
		\leq
		\sup_{0 \leq t \leq T} \diniu^+_t \norm{\sol_t^N-\mfsol_t^N}_\infty
	\end{equation}
	is valid.
	Then, because of \ref{ch4:lemma:right-diff-of-integral} and \ref{ch4:lemma:derive-infinity-norm}, we have that the obtained function is right-differentiable, therefore
	\begin{align*}
		& \sup_{0 \leq t \leq T} \diniu^+_t \norm{\sol_t^N-\mfsol_t^N}_\infty
		\\
		& = \sup_{0 \leq t \leq T} \diniu^+_t \norm{\int_0^t \drift(s, \sol_s^N) - \mfdrift(s, \mfsol_s^N) d s}_\infty
		\\
		&= \sup_{0 \leq t \leq T} \partial^+_t \norm{\int_0^t \drift(s, \sol_s^N) - \mfdrift(s, \mfsol_s^N) d s}_\infty
		\\
			&\leq \sup_{0 \leq t \leq T} \sum_{i=1}^{d \NumParticles_t} \partial^+_t \int_0^t \drift_i(s, \sol_s^N) - \mfdrift_i(s, \mfsol_s^N) d s
			&& {\ref{ch4:lemma:derive-infinity-norm}} 
		\\
			&\leq \sup_{0 \leq t \leq T} \sum_{i=1}^{d \NumParticles_t} 
				\norm{
					\drift_i(t, \sol_t^N) - \mfdrift_i(t, \mfsol_t^N)
				}
			&& \ref{ch4:lemma:right-diff-of-integral} 
		\\
		& \text{(the drift is $0$ for the cAMP)}
		\\
		&= \sup_{0 \leq t \leq T} \sum_{i=1}^{d N} 
		\norm{
			\drift_i(t, \sol_t^N) - \mfdrift_i(t, \mfsol_t^N)
		}
		\\
		& \leq \sup_{0 \leq t \leq T} d N
		\norm{
			\drift(t, \sol_t^N) - \mfdrift(t, \mfsol_t^N)
		}_\infty
		& = d N \sup_{0 \leq t \leq T} \norm{\drift^N(t, \sol_t^N) - \mfdrift(t, \mfsol_t^N)}_\infty
	\end{align*}
	And the proof is concluded.
\end{proof}

\subsection{Generalized attraction}

Only for this section we will use a more general case for the attraction: we will use an additional term which also admits a cut-off Coulomb-type of singularity.


\begin{definition}[Coulomb-type attraction-potential, generalized attraction and drift]\label{ch4:def:generalised-attr&drift}
	Let $\alpha \geq 0$.
	Let $\cutoffSinAttraction = \cutoffSinAttraction (N, \alpha) : \R^d \longrightarrow \R^d$ be a function such that
	\begin{equation}
		0
		\leq 
		\norm{\cutoffSinAttraction}
		\leq 
		K_\curlywedge \left( 
		\frac{1}{\norm{x}} \wedge N^\alpha
		\right)
		=
		\left\{
		\begin{aligned}
			&\frac{K_\curlywedge}{\norm{x}}			&& x \in \{ \norm{x} \geq N^{-\alpha}\}

			&K_\curlywedge\:N^{\alpha}						&& \text{otherwise}
		\end{aligned}
		\right.

			&& 1\leq k \leq N
			\\
			{}^{\genAttraction}\drift_{p,g}(t,x) &:= 0
			&& \text{if } \exists (p,g)
		\end{aligned}
		\right.
	\end{equation}
\end{definition}

For the sake of shortness we will simply write $\drift$ for ${}^{\genAttraction}\drift$ all along this subsection, where such generalised attraction is used.

The first is a simple result of \autocite{peter-boers}, that we simply report here.

\begin{lemma}\label{ch4:lemma:<-1+infty-norm}
	For a function $f$ define the eventually infinite quantities
	\begin{gather*}
		\Norm{f}_{1 \wedge \infty} := 
		\inf_{f_1 + f_\infty = f} 
		\left( 
		\Norm{f_1}_1 + \Norm{f_\infty}_\infty
		\right)
		\\
		\Norm{f}_{1 \vee \infty} := 
		\Norm{f}_1 + \Norm{f}_\infty
	\end{gather*}
	where, if $f$ has a $d$-dimensional domain, we consider $\Norm{f}_p := \max_k \Norm{f_k}$ (here $f_k$ is the $k$-th coordinate of $f$).
	Then for arbitrary functions $f,g$ it is true that
	\begin{equation}
		\Norm{f \star g}_\infty
		\leq
		\Norm{f}_{1 \wedge \infty} \cdot \Norm{g}_{1 \vee \infty}
	\end{equation}
\end{lemma}

\begin{proof}
	Let $g_1 + g_\infty = g$ be a decomposition of $g$.
	Then
	\begin{align*}
		&\Norm{f \star g}_\infty 
		\leq 
		\Norm{f \star g_1}_\infty + \Norm{f \star g_1}_1
		\leq
		\\
		&\leq
		\Norm{f}_\infty \Norm{g_1}_1 + \Norm{f}_1 \Norm{g_\infty}_\infty
		\\
		&\leq
		(\Norm{f}_\infty + \Norm{f}_1) ( \Norm{g_1}_1 + \Norm{g_\infty}_\infty)
		= \Norm{f}_{1 \vee \infty}	( \Norm{g_1}_1 + \Norm{g_\infty}_\infty)
	\end{align*}
	Since this is valid for every decomposition $g_1 + g_\infty = g$, we have our result.
\end{proof}

\subsection{Notation convention}\label{ch4:notation}

From now on, when long calculations will be held, a tilde above $\E{\cdot}, \Prb$ will indicate conditioning with respect to $ (\btime^{p,g})_{p,g}$, and a subscript $k$ will indicate a conditioning with respect to the $k$-th amoeba (\ie $\amba^k$ or $\mfamba^k$, depending on the context):

\begin{align*}
	\ETFoot{}{\cdot}&:=\ECondTimes{\cdot}{}			& 			\PrbTFoot{}{\cdot}&:= \PrbCondTimes{\cdot}{}
	\\
	\ETFoot{k}{\cdot}&:=\ECondTimes{\cdot}{\text{$k$-th amoeba},	}			&				\PrbTFoot{k}{\cdot}&:= \PrbCondTimes{\cdot}{\text{$k$-th amoeba }, }
	\\
	\EFoot{k}{\cdot}&:=\ECond{\cdot}{\text{$k$-th amoeba}}				&				\PrbFoot{k}{\cdot}&:=\PrbCond{\cdot}{\text{$k$-th amoeba}}
\end{align*}

Furthermore, a subscript $\ast$ will mean ambiguity in the presence of the subscript:

\begin{align*}
	\EFoot{\ast}{} &:=  \: \text{either of }	\EFoot{}{\cdot}, \EFoot{k}{\cdot}				&		\PrbFoot{\ast}{\cdot}&:= \: \text{either of }	\PrbFoot{}{\cdot}, \PrbFoot{k}{\cdot}		
	\\
	\ETFoot{\ast}{} &:=  \: \text{either of }	\ETFoot{}{\cdot}, \ETFoot{k}{\cdot}			&		\PrbTFoot{\ast}{\cdot}&:= \: \text{either of }	\PrbTFoot{}{\cdot}, \PrbTFoot{k}{\cdot}		
\end{align*}

Finally we will indicate constants with the symbol $K$, and a subscript will indicate on which mathematical objects such constants depend:

\begin{equation*}
	K_{x,y,z}:= \: \text{generic const $>0$ dependent on $x,y,z$}
\end{equation*}

\subsection{The first law of large numbers}


Now we can finally approach the first law of large numbers.
The proof is classical law of large numbers, following for example \autocite{ana}.
It uses a notation akin to the relabelled model defined in \ref{ch2:model-reformulation}.

\begin{proposition}[The first law of large numbers]\label{ch4:prop:LLN1}
	Let $\hmactive_t$ be the spawning Poisson process of \ref{ch2:model-reformulation}. Let $\mfsol^N$ be a \rv that under $\PrbCondTimes{\cdot}{}$ takes the form
	\begin{equation}
		\mfsol^N = (\mfamba^1, \ldots, \mfamba^N, \mfcamp^1, \ldots, \mfcamp^{\hmactive_t}) \in (\R^d)^{\NumParticles_t} =  (\R^d)^{N + \hmactive_t} 
	\end{equation}
	Let $\drift^{N,k}(t, \cdot)$ be the \emph{generalised} drift of parameter $\alpha\geq 0$ defined in \ref{ch4:def:generalised-attr&drift}.
	Assume that:
	\begin{enumerate}
		\item All the coordinates of $\mfsol^N$ are $\indep$ under the p. measure $\PrbCondTimes{}{}$.
		\item All the coordinates $\mfamba^p$ of $\mfsol^N$ are \iid under the p. measure $\PrbCondTimes{}{}$.
		\item The densities of all coordinates of $\mfsol^N$ are uniformly bounded in $\Lpspace^{\infty}$, \ie
		\begin{equation*}
			\exists K_{dens}>0.\
			\forall p,m.\
			\Norm{\cdens^m}_\infty, \Norm{\adens^p}_\infty < K_{dens}
		\end{equation*}
		where $\adens^p$ and $\cdens^m$ represent the laws of $\mfamba^p$ and $\mfcamp^m$ under $\PrbCondTimes{}{}$ respectively.
	\end{enumerate}
	Fix now $\xi>0$ and define the sets
	\begin{equation*}
		\begin{aligned}
			S^{N,\xi}_\amba
			&:= \left\{ \sup_{1\leq k \leq N} 
				\norm{
					\drift_k^N(t, \mfsol^N) 
					- \ECondTimes{
						\drift^N_k(t,\mfsol^N)
					}{\mfamba^k,}
				}_\infty \geq N^{-\xi} 
			\right\}
			\\
			S^{N, \xi}
			&:= \left\{ \sup_{1\leq k \leq N} 
				\norm{
					\drift_k^N(t, \mfsol^N) 
					- \ECondTimes{
						\drift^N_k(t,\mfsol^N)
					}{}
				}_\infty \geq N^{-\xi} 
			\right\}
		\end{aligned}
	\end{equation*}
	Then, signing as $S_\ast^{N, \xi}$ either of the two sets, for $S_\ast^{N, \xi}$ is valid the laws of large numbers, \ie for all $\gamma>0, \forall n\in\N$ exists $K_n = K_{dens,\attraction, \curlywedge, n}  >0$ (independent from $N, \alpha,\xi$) and $\overline{N}\in N$ such that:
	\begin{align}
		& \forall N> \overline{N}.	&	&\PrbOf{S^{N,\xi}_\ast}{}
			\leq 
			K_n\
			t^n\
			N^{-\gamma}
		&& \label{ch4:LLN1:bound}
	\end{align}
	
	Furthermore, if $\exists 0 < C < 1$ such that $\xi + \alpha < \frac{1- C}{2}$, the subsequent weaker bound is valid $\forall n \in \N$ and $\forall \in \N $:
	
	\begin{align}
		&\PrbOf{S^{N,\xi}_\ast}{}
		\leq 
		K_n\
		t^n\
		N^{-Cn + 1}
		&& \label{ch4:LLN1:weaker-bound}
	\end{align}
\end{proposition}

\begin{remark}
	Recall that if $\drift^N_k$ is the usual drift with the gaussian potential, then one can choose $\alpha = 0$.
\end{remark}

\begin{proof}
	For the sake of shortness, together with the convention at \ref{ch4:notation}, we will use the subsequent notation:
	
	\begin{equation*}
		S^{N, \xi}_\ast := \: \text{either of $S^{N, \xi}_\amba$ and $S^{N, \xi}$}
	\end{equation*}
	
	Notice that 
	
	\begin{align*}
		\PrbOf{S^{N,\xi}_\ast}
		& =\PrbOf{
				\sup_{1\leq k \leq N} 
				\norm{
					\drift_k^N(t, \mfsol^N) 
					- \ETFoot{\ast}{
						\drift^N_k(t,\mfsol^N)

		N \cdot 3 \cdot

		\:

		\PrbOf{\norm{
				\drift^N_{k,i}(t, \mfsol^N)

					\drift^N_{k,i}(t,\mfsol^N)
				}
			} \geq N^{-\xi} 
		}
	}_{\Ch(k,i)}
	\end{align*}
	where $\drift_{k,i}$ indicates the $i$-th coordinate for the $d$-dimensional drift of the $k$-th particle.
	Since the expression has the same form for each coordinate $(k,i)$, we assume that the maximum of $\Ch(k,i)$ is reached in $(1,1)$, and tame with a bound independent from such assumption.
	
	We shall begin by applying Markov inequality with function $f(x):= \norm{x}^{2n}$ for a generic $n \in \N$.
	\begin{align*}
		&=
		3N \cdot

		3
		 N^{\xi 2n + 1} \E{
			\norm{
			\frac{1}{N}

		\\

			\norm{

			(\genAttraction)_1(\mfamba^1 - \mfcamp^m)

			\right]
		}^{2n}
	}
		\\
		&= 3 N^{(\xi -1) 2 n + 1}
		\E{
			\norm{
				\sum _{m = 1}^{\hmactive_t}

					\underbrace{
						(\genAttraction)_1(\mfamba^1 - \mfcamp^m)

					\underbrace{

			}^{2n}
		}
		=
	\end{align*}
	\footnotetext[1]{{Here $(\genAttraction)_1$ indicates the first coordinate of $\genAttraction$, and not a first part of any decomposition of $\genAttraction$.}}
	
	Setting $\Zhi_m := |\Psi_m - \psi_m^\ast|$, we can apply the multinomial formula:
	
	\begin{align}

		\:

		\:

	\end{align}
	
	where $\beta$ is the multi-index $(\beta_1, \ldots, \beta_M)$ and $(\beta)! := \beta_1!\ldots \beta_M!$.
	Then we obtain
	
		\begin{align*}
		&= 3 N^{(\xi -1) 2 n + 1}
		\E{
				\sum_{\beta_1 + \cdots + \beta_{\hmactive_t} =2n} \
				\frac{(2n)!}{(\beta)!}
				\prod_{m=1}^{\hmactive_t}
				\norm{
					\Psi_m - \psi_m^\ast
				}^{\beta_m}
			}
		\\
		& = 3 N^{(\xi -1) 2 n + 1}
		\E{
			\ETFoot{\ast}{
			\sum_{\beta_1 + \cdots + \beta_{\hmactive_t} =2n} \
			\frac{(2n)!}{(\beta)!}
			\prod_{m=1}^{\hmactive_t}
			\norm{
				\Psi_m - \psi_m^\ast
			}^{\beta_m}
		}
		}
		\\
		& = 3 N^{(\xi -1) 2 n + 1}
		\E{
		\sum_{\beta_1 + \cdots + \beta_{\hmactive_t} =2n}
		\frac{(2n)!}{(\beta)!}
		\ETFoot{\ast}{
			\prod_{m=1}^{\hmactive_t}
			\norm{
				\Psi_m - \psi_m^\ast
			}^{\beta_m}
		}}
	\\
	& = 
	3(2n)!\
	N^{(\xi -1) 2 n + 1}
	\E{
		\sum_{\beta_1 + \cdots + \beta_{\hmactive_t} =2n}\
		\ETFoot{\ast}{
			\prod_{m=1}^{\hmactive_t}
			\norm{
				\Psi_m - \psi_m^\ast
			}^{\beta_m}
	}}
=
	\end{align*}
	
	Notice now that for each $m$ the \rvs
	
	\begin{equation}
		 \Psi_m - \psi_m^\ast 
		 \quad
		 = 
		 \quad
		 (\genAttraction)_1(\mfamba^1 - \mfcamp^m)
		- \ETFoot{\ast}{ \left( (\genAttraction)_1 \star \cdens^m \right) (\mfamba^1)}
	\end{equation}
	
	are $\indep$ under $\PrbTFoot{1}{\cdot} = \PrbCondTimes{\cdot}{\mfamba^1,}$, therefore
	
	\begin{equation}\label{proof-of-LLN1:calculations-before-claim}
	\begin{aligned}
		&=
		3(2n)!\
		N^{(\xi -1) 2 n + 1}
		\E{
		\sum_{\beta_1 + \cdots + \beta_{\hmactive_t} =2n} \
		\ETFoot{\ast}{
			\ETFoot{1}{
				\prod_{m=1}^{\hmactive_t}
				\norm{
					\Psi_m - \psi_m^\ast
				}^{\beta_m}
		}} }
	\\
		&=
		3(2n)!\
		N^{(\xi -1) 2 n + 1}
		\E{
	\sum_{\beta_1 + \cdots + \beta_{\hmactive_t} =2n}\
	\prod_{m=1}^{\hmactive_t}
		\ETFoot{\ast}{
			\ETFoot{1}{
				\norm{
					\Psi_m - \psi_m^\ast
				}^{\beta_m}
			}
		}
	}
	\\
	& =
	3(2n)!\
	N^{(\xi -1) 2 n + 1}
	\E{
	\sum_{\beta_1 + \cdots + \beta_{\hmactive_t} =2n}\
	\prod_{m=1}^{\hmactive_t}
	\underbrace{
	\ETFoot{\ast}{
			\norm{
				\Psi_m - \psi_m^\ast
			}^{\beta_m}
		}
	}_{\leq ?}
}
\\
	\end{aligned}
	\end{equation}
	
	\begin{claim}\label{ch4:proof-of-LLN1:claim:bound-on-Psi-psi}
		There exists a constant $K_{\attraction,\text{dens}, \curlywedge}>0$ depending on $\attraction$ and on $K_{\text{dens}}$ such that
		\begin{equation}
			\ETFoot{\ast}{
				\norm{
					\Psi_m - \psi_m^\ast 
				}^{\beta_m}
			}
			\leq 
			\left\{
			\begin{aligned}
				& 1		&&\text{if} \quad \beta_m = 0
				\\
				& 0 		&&\text{if} \quad \beta_m = 1
				\\
				&(K_{\attraction,\text{dens}, \curlywedge} N^{\alpha })^{\beta_m -2}	\quad		&&\text{otherwise}
			\end{aligned}
			\right.
		\end{equation}
	\end{claim}
	
	\begin{proof}[Proof of the claim]
			If $\beta_m =0$ we have
		
		\begin{equation*}
			\ETFoot{\ast}{
				\norm{
					\Psi_m
					- \psi_m^\ast
				}^0
			} = \ETFoot{\ast}{ \ETFoot{1}{ 1} } = 1
		\end{equation*}
		
		If instead $\beta_m =1$ we have:
		\begin{equation*}
			\ETFoot{\ast}{\norm{\Psi_m - \psi_m^\ast }^1 }
			=
			\ETFoot{\ast}{
				\ETFoot{1}{
					\norm{
						\Psi_m
						- \psi_m^\ast
					}
			}}
			= 	\ETFoot{\ast}{
				\norm{
					\Psi_m
					- \ETFoot{\ast}{\Psi_m}
				}
			}	= 0
		\end{equation*}
		
		We need then to take care only of the case $\beta_m \geq 2$.
		Here we have
		
		\begin{align*}
			\ETFoot{\ast}{\norm{\Psi_m - \psi_m^\ast }^{\beta_m} }
			& =
			\ETFoot{\ast}{\norm{\Psi_m - \psi_m^\ast }^{\beta_m - 2} \cdot \norm{\Psi_m - \psi_m^\ast }^{2}}
			\\
			&\leq \Norm{\Psi_m - \psi_m^\ast }^{\beta_m - 2}_\infty \cdot \ETFoot{\ast}{\norm{\Psi_m - \psi_m^\ast }^{2} }
		\end{align*}
		
		Then we can dominate the two terms separately.
		
		\begin{align*}
			\Norm{\Psi_m - \psi_m^\ast }_\infty 
			&= \Norm{(\genAttraction)_1(\mfamba^1 - \mfcamp^m)
				-
				\ETFoot{\ast}{ \left( (\genAttraction)_1 \star \cdens^m \right) (\mfamba^1)}
			}_\infty
			\\
			&=\Norm{\genAttraction}_\infty + \Norm{\genAttraction \star \cdens^m}_\infty
			\\
			& \leq \Norm{\genAttraction}_\infty + \Norm{\genAttraction}_\infty \Norm{\cdens^m}_1
			\\
			& = 2 \Norm{\genAttraction}_\infty = 2 (\Norm{\attraction}_\infty +  \Norm{\cutoffSinAttraction}_\infty )
			\\
			& \leq 2 (2 \Norm{\cutoffSinAttraction}_\infty ) =  K_\curlywedge 4 N^{\alpha}
		\end{align*}
		
		where the last inequality is valid if $N$ is big enough.
		Then
		
		\begin{align*}
			\Norm{\Psi_m - \psi_m^\ast }_\infty^{\beta_m - 2} 
			\leq 
			(K_\curlywedge 4 )^{\beta_m -2} N^{\alpha (\beta_m -2)}
		\end{align*}
		
		Let us now check the other term:
		
		\begin{align*}
			&\ETFoot{\ast}{\norm{\Psi_m - \psi_m^\ast }^{2} } =
			\\
			&= \ETFoot{\ast}{
				\ETFoot{1}{
				\norm{
			(\genAttraction)_1(\mfamba^1 - \mfcamp^m)
			- \psi_m^\ast }^{2}
		}
	}
		\\
		&=
		\ETFoot{\ast}{
		\int
		\left(
			(\genAttraction)_1(\mfamba^1 - x)
			- \psi_m^\ast \right)^{2}
			\:
			\cdens^m(x)
			\:
			dx
		}
		\\
		& = \ETFoot{\ast}{
			\int
			(\genAttraction)_1^2(\mfamba^1 - x)
			\cdens^m(x)
			dx
			- \int 2
			(\genAttraction)_1(\mfamba^1 - x)
			\psi_m^\ast
			\cdens^m(x)
			dx
			+
			(\psi_m^\ast)^2
			 \int
			\cdens^m(x)
			dx
		}
		\\
		&= \ETFoot{\ast}{
			\left( (\genAttraction)_1^2 \star \cdens^m \right) (\mfamba^1)
			- 2 \psi_m^\ast \left( (\genAttraction)_1 \star \cdens^m \right) (\mfamba^1)
			+(\psi_m^\ast)^2
		}
		\\
		& \leq \ETFoot{\ast}{
			\Norm{(\genAttraction)_1^2 \star \cdens^m}_\infty
			+ 2 \psi_m^\ast \Norm{(\genAttraction)_1 \star \cdens^m}_\infty
			+(\psi_m^\ast)^2
		}
		\\
		&= \Norm{(\genAttraction)_1^2 \star \cdens^m}_\infty
		+ 2 \psi_m^\ast \Norm{(\genAttraction)_1 \star \cdens^m}_\infty
		+(\psi_m^\ast)^2 
		\\
		&\leq 
		\Norm{(\genAttraction)_1^2 \star \cdens^m}_\infty
		+ 2\Norm{(\genAttraction)_1 \star \cdens^m}_\infty^2
		+ \Norm{(\genAttraction)_1 \star \cdens^m}_\infty^2
		\\
		& = 
		\Norm{(\genAttraction)_1^2 \star \cdens^m}_\infty
		+ 3\Norm{(\genAttraction)_1 \star \cdens^m}_\infty^2
		\qquad \text{}
		\\
			&\footnotemark[1]
			\leq
			\Norm{\genAttraction^2}_{1 \wedge \infty}
			\Norm{\cdens^m}_{1 \vee \infty}
			+ 3 
			\left(
			\Norm{\genAttraction}_{1 \wedge \infty}
			\Norm{\cdens^m}_{1 \vee \infty}
			\right)^2
			\\
			& =
			K_{\curlywedge,\attraction} (1 + K_{\text{dens}})
			+ 3 K_{\curlywedge,\attraction}^2 (1 + K_{\text{dens}})^2
			= K_{\attraction,\text{dens}, \curlywedge} \leq (K_{\attraction,\text{dens}, \curlywedge})^{\beta_m -2}
		\end{align*}
		\footnotetext[1]{By \ref{ch4:lemma:<-1+infty-norm} and \ref{ch4:lemma:eval-norm-gen-attr}}
		
		Then we have
		
		\begin{align*}
			\ETFoot{\ast}{\norm{\Psi_m - \psi_m^\ast }^{\beta_m} }
			&\leq \Norm{\Psi_m - \psi_m^\ast }^{\beta_m - 2}_\infty \cdot \ETFoot{\ast}{\norm{\Psi_m - \psi_m^\ast }^{2} }
			\\
			&\leq (4N^\alpha K_{\attraction,\text{dens}, \curlywedge})^{\beta_m -2} = (K_{\attraction,\text{dens}, \curlywedge} N^\alpha)^{\beta_m -2} 
		\end{align*}
		And this concludes the proof of the claim.
	\end{proof}
	
	Now we finally obtained a bound independent from the coordinate $(1,1)$ of the drift, thus, resuming with calculations from \ref{proof-of-LLN1:calculations-before-claim} and exploiting the claim \ref{ch4:proof-of-LLN1:claim:bound-on-Psi-psi}, we have that
	
	\begin{align*}
		\PrbOf{S^{N,\xi}_\ast}
		& =\PrbOf{
			\sup_{1\leq k \leq N} 
			\norm{
				\drift_k^N(t, \mfsol^N) 
				- \ETFoot{\ast}{
					\drift^N_k(t,\mfsol^N)
				}
			}_\infty \geq N^{-\xi} 
		}
		\\
		&\leq
		N \cdot 3 \cdot
		\max_{k,i}
		\:
			\PrbOf{\norm{
					\drift^N_{k,i}(t, \mfsol^N) 
					- \ETFoot{\ast}{
						\drift^N_{k,i}(t,\mfsol^N)
					}
				} \geq N^{-\xi} 
			}
		\\
		&\leq
		3(2n)!\
		N^{(\xi -1) 2 n + 1}
		\E{
			\sum_{\beta_1 + \cdots + \beta_{\hmactive_t} =2n}\
			\prod_{m=1}^{\hmactive_t}
				\ETFoot{\ast}{
					\norm{
						\Psi_m - \psi_m^\ast
					}^{\beta_m}
				}
			}
			\\
		&\leq
		3(2n)!\
		N^{(\xi -1) 2 n + 1}
		\E{
			\sum_{\beta_1 + \cdots + \beta_{\hmactive_t} =2n}\
				\prod_{m=1}^{\hmactive_t}
				\left(
				\1_{\beta_m = 0} + \1_{\beta_m \geq 2}(
				\underbrace{
					K_{\attraction,\text{dens}, \curlywedge} N^\alpha
				}_{=:\Iu}
					)^{\beta_m -2} 
				\right)
		}
	\end{align*}
	
	We can assume that $\Iu \geq 1$ (eventually modulate the value $K_{\attraction,\text{dens}, \curlywedge} $ accordingly).
	
	\begin{claim}\label{ch4:proof-of-LLN1:sum-prod-Iu-inequality}
		$\Iu \geq 1$ is independent from the multi-index $\beta$, therefore
		\begin{equation}
			\sum_{\beta_1 + \cdots + \beta_M =2n}\
			\prod_{m=1}^{M}\
			\left(
			\1_{\beta_m = 0} + \1_{\beta_m \geq 2}
			\Iu^{\beta_m -2} 
			\right)
			\leq
			(2n)^n\
			(M)^n\
			\Iu^{2n-2}
		\end{equation}
	\end{claim}
	
	\begin{proof}[Proof of the claim]
		First of all notice that if the multi-index $\beta$ has an entry $\beta_m =1$, then the whole product under $\prod_{m=1}^{M}$ goes to $0$.
		Besides, if $\beta_m=0$, then the relative factor is $1$ and therefore does not influence the product.
		In equations:
		
		\begin{align*}
			&\prod_{m=1}^M
			\left(
			\1_{\beta_m = 0} + \1_{\beta_m \geq 2} \Iu^{\beta_m -2} 
			\right) = 
			\prod_{
				\substack{
					m=1
					\\
					\beta_m \neq 0
				}
			}^M
				\left(
					\1_{\beta_m \geq 2} \Iu^{\beta_m -2} 
				\right)
			\
			= 
			\
			\prod_{
				\substack{
					m=1
					\\
					\beta_m \neq 0
				}
			}^M
			(\1_{\beta_m \neq 1} \Iu^{\beta_m -2} )
			=
			\\
			&\qquad
			=
			\1_{\beta_m \neq 1}
			\prod_{
				\substack{
					m=1
					\\
					\beta_m \neq 0
				}
			}^M
			\Iu^{\beta_m}
			\:
			\cdot
			\:
			\prod_{
				\substack{
					m=1
					\\
					\beta_m \neq 0
				}
			}^M
			\Iu^{-2}\
			=			
			\1_{\beta_m \neq 1}
			\Iu^{\sum_{\beta_m \neq 0} \beta_m}
			\Iu^{-2\sum_{\beta_m \neq 0} 1} 
		\end{align*}
		
		Therefore:
		
		\begin{equation*}
			\begin{aligned}
				\sum_{\beta_1 + \cdots + \beta_M =2n}\ \prod_{m=1}^{M}
				\left(
				\1_{\beta_m = 0} + \1_{\beta_m \geq 2}
				\Iu^{\beta_m -2} 
				\right)
				&=
				\sum_{\beta_1 + \cdots + \beta_{M} =2n}\
				\1_{\beta_m \neq 1}
				\Iu^{\sum_{\beta_m \neq 0} \beta_m}
				\Iu^{-2\sum_{\beta_m \neq 0} 1} 
				\\
