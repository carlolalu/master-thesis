\begin{proposition}
	Let $\sol_t^N$ be the solution to the model \ref{ch2:model} and let $\mfsol_t^N$ be the solution to the mean-field model \ref{ch3:MF-particles}, each with $N$ amoebas.
	Then 
	\begin{equation}
		\PrbOf{  
			\sup_{0 \leq t \leq T}
			\Norm{
				\sol_t^N - \mfsol_t^N
			}_\infty 
			\geq
			N^{-\cutparameter}
		} \leq 
		[[write here the proper bound]]
	\end{equation}
\end{proposition}

\section{Proof}

The proof is based on the usage of the Groenwall argument on the auxiliary process 

\begin{equation}
	\auxProc_T^N := 1 \wedge \left( 
	N^{\cutparameter}
	\sup_{0 \leq t \leq T}
	\Norm{
		\sol_t^N - \mfsol_t^N
	}_\infty \right)
\end{equation}

In fact we have that

	\begin{align*}
	\PrbOf{  
		\sup_{0 \leq t \leq T}
		\Norm{
			\sol_t^N - \mfsol_t^N
		}_\infty 
		\geq
		N^{-\cutparameter}
	}
	&=
	\PrbOf{  
		N^{\cutparameter}\:
		\sup_{0 \leq t \leq T}
		\Norm{
			\sol_t^N - \mfsol_t^N
		}_\infty 
		\geq
		1
	}
	\\
	&=
	\PrbOf{  
		\Big(
		N^{\cutparameter}\:
		\sup_{0 \leq t \leq T}
		\Norm{
			\sol_t^N - \mfsol_t^N
		}_\infty 
		\Big)
		\wedge
		1
		\geq
		1
	}
	\\
	&=
	\PrbOf{  
		\auxProc_t^N
		\geq
		1
	}\
	\leq\
	\E{ 
		\auxProc_t^N
	} 
\end{align*}

where in the latter inequality we applied Markov with $f(x) = \norm{x}$.
