\begin{proposition}
	Let $\sol_t^N$ be the solution to the model \ref{ch2:model} and let $\mfsol_t^N$ be the solution to the mean-field model \ref{ch3:MF-particles}, each with $N$ amoebas.
	Let $T > 0$ and let $0< \delta <\xi <\frac{1}{2}$.
	Then there exists an $\overline{N}>0$ and an increasing continuous function $F^{\uparrow}: \R_+ \longrightarrow \R_+$ independent from $\overline{N},N$ such that for every $N > \overline{N}$ we have that
	\begin{equation}
		\PrbOf{  
			\sup_{0 \leq t \leq T}
			\Norm{
				\sol_t^N - \mfsol_t^N
			}_\infty 
			\geq
			N^{-\cutparameter}
		}
		\leq\
		F^{\uparrow}(T)\
		N^{-(\xi-\delta)}
	\end{equation}
\end{proposition}

The proof is based on the usage of the Groenwall argument on the auxiliary process 

\begin{equation}
	\auxProc_T^N := 1 \wedge \left( 
	N^{\cutparameter}
	\sup_{0 \leq t \leq T}
	\Norm{
		\sol_t^N - \mfsol_t^N
	}_\infty \right)
\end{equation}

In fact we have that

	\begin{align*}
	\PrbOf{  
		\sup_{0 \leq t \leq T}
		\Norm{
			\sol_t^N - \mfsol_t^N
		}_\infty 
		\geq
		N^{-\cutparameter}
	}
	&=
	\PrbOf{  
		N^{\cutparameter}\:
		\sup_{0 \leq t \leq T}
		\Norm{
			\sol_t^N - \mfsol_t^N
		}_\infty 
		\geq
		1
	}
	\\
	&=
	\PrbOf{  
		\Big(
		N^{\cutparameter}\:
		\sup_{0 \leq t \leq T}
		\Norm{
			\sol_t^N - \mfsol_t^N
		}_\infty 
		\Big)
		\wedge
		1
		\geq
		1
	}
	\\
	&=
	\PrbOf{  
		\auxProc_T^N
		\geq
		1
	}\
	\leq\
	\E{ 
		\auxProc_T^N
	} 
\end{align*}

where in the latter inequality we applied Markov with $f(x) = \norm{x}$.
Then we need to find a bound for $\E{\auxProc_T^N} $.

\section{Stochastic estimate with Groenwall}

Bear in mind that the functions involved are not continuous but \cadlag, thus neither the classic Groenwall argument nor the one generalised for dini derivatives \cite[Theorem 1.4.1]{groenwall-with-dini-derivatives} can be applied.
We will instead of that use the one from \cite[Chapter 1, Theorem 8.1]{groenwall-non-continuous-functions}:

\begin{lemma}
	Let $T >0, c>0$. Let $u(\cdot)$ be a Borel-measurable bounded non-negative function on $[0,S]$ and let $v(\cdot)$ be a non-negative integrable function in $[0,S]$.
	If for all $T \in [0,S]$
		\begin{equation*}
			u(T) \leq \int_0^T u(t) v(t) dt + c
		\end{equation*}
	then for all $T \in [0,S]$
			\begin{equation*}
				u(T) \leq c\int_0^T \exp(v(t) )dt
			\end{equation*}
\end{lemma}

Consider $0 < t \leq T$ and calculate then:

\begin{align*}
	\diniu^+_t \auxProc_t^N

Now observe that if $N^{\cutparameter}\norm{\sol_t^N - \mfsol_t^N}_\infty \geq 1$ then $\auxProc_t^N =1$, which is the maximum value that $\auxProc_t^N$ can reach.
This means that in such case we have $\diniu^+_t\auxProc_t^N \leq 0$, because the function, in the near future either remains constant or decreases.
This means that

\begin{lemma}
	$\PrbOf{\int_0^S 
		\norm{\drift^N(t, \mfsol_t^N) - \mfdrift^N(t, \mfsol_t^N)}_\infty 
		dt = 0} = 0 $
\end{lemma}

\begin{proof}
	Again we proceed similarly to \ref{ch4:only-one-amoeba-will-survive}: we just need to prove that the result holds with respect to $\PrbCondTimes{}{}$.
	First notice that
	
	\begin{align*}
		&\PrbCondTimes{\int_0^S
			\norm{\drift^N(t, \mfsol_t^N) - \mfdrift^N(t, \mfsol_t^N)}_\infty 
			dt = 0}{}=
			\\
		&= \PrbCondTimes{
			\forall p,i.\
			\int_0^S
			\norm{\drift_{p;i}^N(t, \mfsol_t^N) - \mfdrift_{p;i}^N(t, \mfsol_t^N)}
			dt
			= 0
			}{}
		\\
		&\leq
		\sum_{p,i}\
		\PrbCondTimes{ \int_0^S
			\norm{\drift_{p;i}^N(t, \mfsol_t^N) - \mfdrift_{p;i}^N(t, \mfsol_t^N)}
			dt
			}{}
		\\
		&=
		\sum_{p,i}\
		\int
		\1_{\{0\}}
		\Big(
		\underbrace{
			\int_0^S
				\norm{\drift_{p;i}^N(t, x) - \mfdrift_{p;i}^N(t, x)}
			dt
		}_{\Zhi_S(x,y)}
		\Big)
		\cdot
		\PrbCondTimes{\mfsol_t^N =dx}{}
	\end{align*}
	
	Let us show that $\left\{
	\Zhi_S = 0
	\right\}$ has measure $0$.
	Here we will write "$\forall^\lebsge t \in [0,S]$" with the meaning of "for almost every $t \in [0,S]$ with respect to the measure $\lebsge$".
	
	\begin{align*}
		x \in \left\{		\Zhi_S = 0		\right\} \quad 
		&\Longleftrightarrow\
		\forall^\lebsge t \in [0,S].\quad
			\norm{\drift_{p;i}^N(t, x) - \mfdrift_{p;i}^N(t, x)}
			= 0
			\\
		&\Longleftrightarrow\
		\forall^\lebsge t \in [0,S].\quad
			\frac{1}{N} \sum _{m = 1}^{\hmactive_t} \attraction(x^p - x^{N+m}) 
			 - \lambda \left(\attraction_i \star \int_0^t \cdens_t^\tau 	d \tau \right) (x^{p})
		= 0
		\\
		&\Longleftrightarrow\
		\forall^\lebsge t \in [0,S].\quad
					\frac{1}{N} \sum _{m = 1}^{\hmactive_t} \left(\attraction \star \delta_{x^{N+m}}\right) (x^p ) 
			- \left(\attraction_i \star \lambda \int_0^t \cdens_t^\tau 	d \tau \right) (x^{p})
		= 0
		\\
		&\Longleftrightarrow\
		\forall^\lebsge t \in [0,S].\quad
		\left[\attraction \star 
		\left(
		\frac{1}{N} \sum _{m = 1}^{\hmactive_t} \delta_{x^{N+m}}
		- \lambda \int_0^t \cdens_t^\tau 	d \tau \right) 
		\right]
		(x^{p})
		= 0
		\\
		&\Longleftrightarrow\
		\forall^\lebsge t \in [0,S].\quad
		\frac{1}{N} \sum _{m = 1}^{\hmactive_t} \delta_{x^{N+m}}
		= \lambda \int_0^t \cdens_t^\tau 	d \tau
	\end{align*}
	
	Now observe that if $\hmactive_t = 0$, the last equality is trivially not satisfied because its \lhs is $0$ while its \rhs is positive.
	If instead $\hmactive_t \neq 0$ we would have a pure distribution on the \lhs and $\Lpspace^1$ function on the \rhs, which is of course also not attainable.
	Then $ \left\{	\Zhi_S = 0	\right\}$ is the empty set, and the thesis is proved.	
\end{proof}

\end{equation*}

		\1_{\setB_t^c}\
		\exp \left(
		TK_\attraction\
		\frac{\hmactive_T}{N}\
		\right)
	}\
	dt +
	\\
	&\qquad
	+ N^{\cutparameter}\
	K_\attraction\
	\lambda\
	S\
	\int_0^S
	\E{
		\1_{\setB_t^c}\
		\exp \left(
		TK_\attraction\
		\frac{\hmactive_T}{N}\
		\right)
	}
	dt
	\\
	&\leq
	N^{\cutparameter-\xi}\
	K\
	S\
	\E{
		\exp \left(
		\frac{TK_\attraction}{N}\
		\hmactive_T
		\right)
	} +
	\\
	&\qquad
	+ N^{\cutparameter}\
	K_\attraction\

	N^{-1}
\E{
	\left(\hmactive_S
	\right)^p
}^{\frac{1}{p}}

	\exp \left(

	&\qquad

\E{
	\exp \left(
	\frac{vTK_\attraction}{N}\
	\hmactive_T
	\right)
}^{\frac{1}{v}}
	dt
\end{align}

Now we can evaluate the probability of $\setB_t^c$ with the law of large numbers (\ref{ch5:LLN}): for each $n\in \N$ there is some $\overline{N}$ such that for every $N \geq \overline{N}$ we have that

\begin{equation*}
	\PrbOf{\setB_t^c}\ 
	\leq 
	K_n\
	(t \vee 1)^{4n}\
	N^{-K_\xi n+ 1}
\end{equation*}

for some constant $K_\xi$.
The higher moments of $\hmactive_T$ can be evaluated with \ref{app:Poisson-moments} obtaining (for $N$ big enough, so that the Poisson parameter is $\geq 1$)

\begin{equation*}
	\E{
		\left(\hmactive_S
		\right)^p
	} \leq
	f^{\uparrow}(p) (\lambda N T)^p
\end{equation*}

Finally we can notice that for every $a$ the expectation values of the exponentials $\E{
	\exp \left(
	\frac{a}{N}\
	\hmactive_T
	\right) }$
is nothing else than an evaluation in $\frac{a}{N}$ of the moment generating function of the Poisson random variable $\hmactive_t$. By the result of \ref{app:Poisson-generating-function} such quantities are:

\begin{align*}
	\E{
		\exp \left(
	\frac{a}{N}\
	\hmactive_T
	\right) }
	= \exp\left[\lambda N T (e^{\frac{a}{N}}-1)\right]
\end{align*}

Then we can use De l'Hopital to see that:

\begin{align*}
	\lim_N\
	\exp\left[\lambda N T (e^{\frac{a}{N}}-1)\right]
	&=
	\exp\left[
	\lim_N\
	\frac{e^{\frac{a}{N}}-1}{
		\left(\frac{1}{\lambda N T } \right)}
	\right]
	&
	=
	\exp\left[
	\lim_N\
	\frac{e^{\frac{a}{N}}(-1)\frac{a}{N^2}}{
		\left(\frac{-1}{\lambda TN^2 } \right)}
	\right]	
	= \exp\left[
	a\lambda T e^0
	\right]	=e^{a\lambda T}
\end{align*}

This means that if we choose $\epsilon >0$ than there exists $\overline{N}= \overline{N}_{a, \epsilon} >0$ such that, if $N > \overline{N}$, then 

\begin{equation*}
	\E{
		\exp \left(
		\frac{a}{N}\
		\hmactive_T
		\right) } \leq e^{a\lambda T} + \epsilon
\end{equation*}

With these estimates in mind we can go on from \ref{ch6:pre-final-evaluation-on-avg-J} to calculate that, for $N$ big enough we have

\begin{align*}
	\E{\auxProc_T^N} 
	&\leq
	N^{\cutparameter-\xi}\
	K\
	S\
	(e^{K_\attraction\lambda T^2} + \epsilon)+
	\\
	&\qquad
	+ N^{\cutparameter}\
	K_\attraction\
	\int_0^S\
	N^{-1}
	\left[f^{\uparrow}(p) (\lambda N T)^p\right]^{\frac{1}{p}}\
		\left(
		K_n\
		(t \vee 1)^{4n}\
		N^{-K_\xi n+ 1}
		\right)\
		(e^{qvK_\attraction \lambda T^2} + \epsilon)^\frac{1}{qv}
	dt +
	\\
	&\qquad
	+ N^{\cutparameter}\
	K_\attraction\
	\lambda\
	S\
	\int_0^S
		\left(
		K_n\
		(t \vee 1)^{4n}\
		N^{-K_\xi n+ 1}
		\right)\
		(e^{vK_\attraction \lambda T^2} + \epsilon)^\frac{1}{v}
	dt
	\\
	& =
	N^{\cutparameter-\xi}\
	K\
	S\
	(e^{K_{\attraction,\lambda} T^2} + \epsilon)
	\\
	&\qquad
	+
	N^{\cutparameter}\
	K_\attraction\
	\int_0^S\
	N^{-1}N\
	K_{p,\lambda,n}\
	T\
	(t \vee 1)^{4n}\
	N^{-K_\xi n+ 1}
	(e^{K_{q,v,\attraction,\lambda} T^2} + \epsilon)^\frac{1}{qv}
	dt +
	\\
	&\qquad
	+ N^{\cutparameter}\
	K_{\attraction,n,\lambda}\
	S\
	\int_0^S
	(t \vee 1)^{4n}\
	N^{-K_\xi n+ 1}\
	(e^{K_{v,\attraction,\lambda} T^2} + \epsilon)^\frac{1}{v}
	dt \leq
\end{align*}

then, since the only term in the integral which is depending on $t$ is $(t \vee 1)^{4n}\leq (S \vee 1)^{4n}$, we do get

\begin{align*}
	&\leq
	N^{\cutparameter-\xi}\
	K\
	S\
	(e^{K_{\attraction,\lambda} T^2} + \epsilon)
	\\
	&\qquad
	+
	N^{\cutparameter-K_\xi n+ 1}\
	K_{p,\lambda,n,\attraction}\
	T\
	(S \vee 1)^{4n}\
	(e^{K_{q,v,\attraction,\lambda} T^2} + \epsilon)^\frac{1}{qv}
	\int_0^S\
	dt +
	\\
	&\qquad
	+ N^{\cutparameter-K_\xi n+ 1}\
	K_{\attraction,n,\lambda}\
	S\
	(S \vee 1)^{4n}\
	(e^{K_{v,\attraction,\lambda} T^2} + \epsilon)^\frac{1}{v}
	\int_0^S
	dt 
	\\
	&\leq
	N^{\cutparameter-\xi}\
	K\
	S\
	(e^{K_{\attraction,\lambda} T^2} + \epsilon)
	\\
	&\qquad
	+
	N^{\cutparameter-K_\xi n+ 1}\
	K_{p,\lambda,n,\attraction}\
	S^2\
	(S \vee 1)^{4n}\
	(e^{K_{q,v,\attraction,\lambda} T^2} + \epsilon)^\frac{1}{qv}
	+
	\\
	&\qquad
	+ N^{\cutparameter-K_\xi n+ 1}\
	K_{\attraction,n,\lambda}\
	S^2\
	(S \vee 1)^{4n}\
	(e^{K_{v,\attraction,\lambda} T^2} + \epsilon)^\frac{1}{v}
	\\
	&\leq 
	N^{\cutparameter-\xi}\

	S\

	N^{\cutparameter-K_\xi n+ 1}\
	K_{p,\lambda,n,\attraction}\

	2\

\end{align*}

Choose now $n$ so that $N^{\cutparameter-K_\xi n+ 1} < N^{\cutparameter-\xi}$. Then

\begin{align*}
	&\leq

	\left\{

\begin{equation*}
	\E{\auxProc_T^N}

\begin{equation*}
	\E{\auxProc_T^N}
	\leq

Define now the continuous increasing function

\begin{equation*}
	F^{\uparrow}(T) := K_{p,\lambda,n,\attraction}\
	(T \vee 1)^{4n+2}\

\end{equation*}

to obtain

\begin{equation*}
	\E{\auxProc_T^N}
	\leq
	F^{\uparrow}(T)\
	N^{\cutparameter-\xi}\
	=\
	F^{\uparrow}(T)\
	N^{-(\xi - \cutparameter)}\
\end{equation*}

And thus the main result has been proven.
